101 def _process_one_node(self, node: TabContainer):
102 # There is no existing tab set. Let's start a new one.
103 if not self.stack:

105 return
106  
107 # There should never be an empty "current" tab set.
108 assert self.stack[-1]
109  
110 close_till = None
111 append = False
112 for tab_set in reversed(self.stack[:]):
113 last_node = tab_set[-1]
114  
115 # Is this node a direct child of the last node in this tab-set?
116 is_child = node in last_node.children[1]
117 if is_child:
118 close_till = tab_set
119 append = False
120 break
121  
122 # Is this node a sibling of the last node in this tab-set?
123 is_sibling = (
