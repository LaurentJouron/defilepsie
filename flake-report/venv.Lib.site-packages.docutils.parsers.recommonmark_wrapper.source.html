77 def parse(self, inputstring, document):
78 """Use the upstream parser and clean up afterwards.
79 """
80 # check for exorbitantly long lines
81 for i, line in enumerate(inputstring.split('\n')):
82 if len(line) > document.settings.line_length_limit:
83 error = document.reporter.error(
  • E228 Missing whitespace around modulo operator
84 'Line %d exceeds the line-length-limit.'%(i+1))
85 document.append(error)
86 return
87  
88 # pass to upstream parser
89 try:
90 CommonMarkParser.parse(self, inputstring, document)
91 except Exception as err:
92 if document.settings.traceback:
93 raise err
94 error = document.reporter.error('Parsing with "recommonmark" '
  • E228 Missing whitespace around modulo operator
95 'returned the error:\n%s'%err)
96 document.append(error)

98 # Post-Processing
99 # ---------------
100  
101 # merge adjoining Text nodes:
102 for node in document.findall(nodes.TextElement):
103 children = node.children
104 i = 0
105 while i+1 < len(children):
106 if (isinstance(children[i], nodes.Text)
  • E129 Visually indented line with same indent as next logical line
107 and isinstance(children[i+1], nodes.Text)):
108 children[i] = nodes.Text(children[i]+children.pop(i+1))
109 children[i].parent = node
110 else:
111 i += 1
112  
113 # add "code" class argument to literal elements (inline and block)
114 for node in document.findall(is_literal):
115 if 'code' not in node['classes']:
116 node['classes'].append('code')
117 # move "language" argument to classes
118 for node in document.findall(nodes.literal_block):
119 if 'language' in node.attributes:
120 node['classes'].append(node['language'])
121 del node['language']
122  
123 # replace raw nodes if raw is not allowed
124 if not document.settings.raw_enabled:
125 for node in document.findall(nodes.raw):
126 warning = document.reporter.warning('Raw content disabled.')
127 node.parent.replace(node, warning)
128  
129 # drop pending_xref (Sphinx cross reference extension)
130 for node in document.findall(addnodes.pending_xref):
131 reference = node.children[0]
132 if 'name' not in reference:
133 reference['name'] = nodes.fully_normalize_name(
134 reference.astext())
135 node.parent.replace(node, reference)
