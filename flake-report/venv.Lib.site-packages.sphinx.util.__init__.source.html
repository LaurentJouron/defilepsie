198 def parselinenos(spec: str, total: int) -> list[int]:
199 """Parse a line number spec (such as "1,2,4-6") and return a list of
200 wanted line numbers.
201 """
202 items = []
203 parts = spec.split(',')
204 for part in parts:
205 try:
206 begend = part.strip().split('-')
207 if begend == ['', '']:
208 raise ValueError
209 if len(begend) == 1:
210 items.append(int(begend[0]) - 1)
211 elif len(begend) == 2:
212 start = int(begend[0] or 1) # left half open (cf. -10)
  • E501 Line too long (86 > 79 characters)
213 end = int(begend[1] or max(start, total)) # right half open (cf. 10-)
214 if start > end: # invalid range (cf. 10-1)
215 raise ValueError
216 items.extend(range(start - 1, end))
217 else:
218 raise ValueError
219 except ValueError as exc:
220 msg = f'invalid line number spec: {spec!r}'
221 raise ValueError(msg) from exc

247 def encode_uri(uri: str) -> str:
248 split = list(urlsplit(uri))
249 split[1] = split[1].encode('idna').decode('ascii')
250 split[2] = quote_plus(split[2].encode(), '/')
251 query = [(q, v.encode()) for (q, v) in parse_qsl(split[3])]
252 split[3] = urlencode(query)
253 return urlunsplit(split)
