1 """Extension to save typing and prevent hard-coding of base URLs in reST files.
2  
3 This adds a new config value called ``extlinks`` that is created like this::
4  
5 extlinks = {'exmpl': ('https://example.invalid/%s.html', caption), ...}
6  
7 Now you can use e.g. :exmpl:`foo` in your documents. This will create a
8 link to ``https://example.invalid/foo.html``. The link caption depends on
9 the *caption* value given:
10  
11 - If it is ``None``, the caption will be the full URL.
12 - If it is a string, it must contain ``%s`` exactly once. In this case the
13 caption will be *caption* with the role content substituted for ``%s``.
14  
15 You can also give an explicit caption, e.g. :exmpl:`Foo <foo>`.
16  
17 Both, the url string and the caption string must escape ``%`` as ``%%``.
18 """

45 class ExternalLinksChecker(SphinxPostTransform):
46 """
47 For each external link, check if it can be replaced by an extlink.
48  
  • E501 Line too long (88 > 79 characters)
49 We treat each ``reference`` node without ``internal`` attribute as an external link.
50 """
51  
52 default_priority = 500
53  
54 def run(self, **kwargs: Any) -> None:
55 if not self.config.extlinks_detect_hardcoded_links:
56 return
57  
58 for refnode in self.document.findall(nodes.reference):
59 self.check_uri(refnode)
60  
61 def check_uri(self, refnode: nodes.reference) -> None:
62 """
63 If the URI in ``refnode`` has a replacement in ``extlinks``,
64 emit a warning with a replacement suggestion.
65 """
66 if 'internal' in refnode or 'refuri' not in refnode:
67 return
68  
69 uri = refnode['refuri']
70 title = refnode.astext()
71  
72 for alias, (base_uri, _caption) in self.app.config.extlinks.items():
  • E501 Line too long (88 > 79 characters)
73 uri_pattern = re.compile(re.escape(base_uri).replace('%s', '(?P<value>.+)'))
74  
75 match = uri_pattern.match(uri)
76 if (
77 match and
78 match.groupdict().get('value') and
79 '/' not in match.groupdict()['value']
80 ):
81 # build a replacement suggestion
82 msg = __('hardcoded link %r could be replaced by an extlink '
83 '(try using %r instead)')
84 value = match.groupdict().get('value')
85 if uri != title:
86 replacement = f":{alias}:`{rst.escape(title)} <{value}>`"
87 else:
88 replacement = f":{alias}:`{value}`"
89 logger.warning(msg, uri, replacement, location=refnode)

92 def make_link_role(name: str, base_url: str, caption: str) -> RoleFunction:
93 # Check whether we have base_url and caption strings have an '%s' for
94 # expansion. If not, fall back the the old behaviour and use the string as
95 # a prefix.
96 # Remark: It is an implementation detail that we use Pythons %-formatting.
97 # So far we only expose ``%s`` and require quoting of ``%`` using ``%%``.
98 def role(typ: str, rawtext: str, text: str, lineno: int,
99 inliner: Inliner, options: dict[str, Any] | None = None,
100 content: Sequence[str] = (),
101 ) -> tuple[list[Node], list[system_message]]:
102 text = utils.unescape(text)
103 has_explicit_title, title, part = split_explicit_title(text)
104 full_url = base_url % part
105 if not has_explicit_title:
106 if caption is None:
107 title = full_url
108 else:
109 title = caption % part
110 pnode = nodes.reference(title, title, internal=False, refuri=full_url)
111 return [pnode], []
112 return role

116 for name, (base_url, caption) in app.config.extlinks.items():
