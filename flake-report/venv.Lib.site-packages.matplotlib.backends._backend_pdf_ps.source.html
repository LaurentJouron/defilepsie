22 def get_glyphs_subset(fontfile, characters):
23 """
24 Subset a TTF font
25  
26 Reads the named fontfile and restricts the font to the characters.
27 Returns a serialization of the subset font as file-like object.
28  
29 Parameters
30 ----------
31 fontfile : str
32 Path to the font file
33 characters : str
34 Continuous set of characters to include in subset
35 """
36  
37 options = subset.Options(glyph_names=True, recommended_glyphs=True)
38  
39 # Prevent subsetting extra tables.
40 options.drop_tables += [
41 'FFTM', # FontForge Timestamp.
42 'PfEd', # FontForge personal table.
43 'BDF', # X11 BDF header.
  • E501 Line too long (88 > 79 characters)
44 'meta', # Metadata stores design/supported languages (meaningless for subsets).
45 ]
46  
47 # if fontfile is a ttc, specify font number
48 if fontfile.endswith(".ttc"):
49 options.font_number = 0
50  
51 with subset.load_font(fontfile, options) as font:
52 subsetter = subset.Subsetter(options=options)
53 subsetter.populate(text=characters)
54 subsetter.subset(font)
55 fh = BytesIO()
56 font.save(fh, reorderTables=False)
57 return fh
