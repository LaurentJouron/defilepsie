64 def guess_mimetype(
65 filename: PathLike[str] | str = '',
66 default: str | None = None,
67 ) -> str | None:
68 ext = path.splitext(filename)[1].lower()
69 if ext in mime_suffixes:
70 return mime_suffixes[ext]
71 if path.exists(filename):
72 try:
73 imgtype = _image_type_from_file(filename)
74 except ValueError:
75 pass
76 else:
77 return 'image/' + imgtype
78 return default

85 def parse_data_uri(uri: str) -> DataURI | None:
86 if not uri.startswith('data:'):
87 return None
88  
89 # data:[<MIME-type>][;charset=<encoding>][;base64],<data>
90 mimetype = 'text/plain'
91 charset = 'US-ASCII'
92  
93 properties, data = uri[5:].split(',', 1)
94 for prop in properties.split(';'):
95 if prop == 'base64':
96 pass # skip
97 elif prop.startswith('charset='):
98 charset = prop[8:]
99 elif prop:
100 mimetype = prop
101  
102 image_data = base64.b64decode(data)
103 return DataURI(mimetype, charset, image_data)

106 def _image_type_from_file(filename: PathLike[str] | str) -> str:
107 with open(filename, 'rb') as f:
108 header = f.read(32) # 32 bytes
109  
110 # Bitmap
111 # https://en.wikipedia.org/wiki/BMP_file_format#Bitmap_file_header
112 if header.startswith(b'BM'):
113 return 'bmp'
114  
115 # GIF
116 # https://en.wikipedia.org/wiki/GIF#File_format
117 if header.startswith((b'GIF87a', b'GIF89a')):
118 return 'gif'
119  
120 # JPEG data
121 # https://en.wikipedia.org/wiki/JPEG_File_Interchange_Format#File_format_structure
122 if header.startswith(b'\xFF\xD8'):
123 return 'jpeg'
124  
125 # Portable Network Graphics
126 # https://en.wikipedia.org/wiki/PNG#File_header
127 if header.startswith(b'\x89PNG\r\n\x1A\n'):
128 return 'png'
129  
130 # Scalable Vector Graphics
131 # https://svgwg.org/svg2-draft/struct.html
132 if b'<svg' in header.lower():
133 return 'svg+xml'
134  
135 # TIFF
136 # https://en.wikipedia.org/wiki/TIFF#Byte_order
137 if header.startswith((b'MM', b'II')):
138 return 'tiff'
139  
140 # WebP
141 # https://en.wikipedia.org/wiki/WebP#Technology
142 if header.startswith(b'RIFF') and header[8:12] == b'WEBP':
143 return 'webp'
144  
145 msg = 'Could not detect image type!'
146 raise ValueError(msg)
