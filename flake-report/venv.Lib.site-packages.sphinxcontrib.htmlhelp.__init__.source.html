78 def chm_htmlescape(s: str, quote: bool = True) -> str:
79 """
80 chm_htmlescape() is a wrapper of html.escape().
81 .hhc/.hhk files don't recognize hex escaping, we need convert
82 hex escaping to decimal escaping. for example: ``&#x27;`` -> ``&#39;``
83 html.escape() may generates a hex escaping ``&#x27;`` for single
84 quote ``'``, this wrapper fixes this.
85 """
86 s = html.escape(s, quote)
87 s = s.replace('&#x27;', '&#39;') # re-escape as decimal
88 return s
