494 def _resolved_interpreter(self) -> str:

496  
497 Based heavily on tox's implementation (tox/interpreters.py).

500 # Nox.

503

506  
507 if self.interpreter is None:
508 self._resolved = sys.executable
509 return self._resolved
510  
511 # Otherwise we need to divine the path to the interpreter. This is
512 # designed to accept strings in the form of "2", "2.7", "2.7.13",
513 # "2.7.13-32", "python2", "python2.4", etc.

517 # If this is just a X, X.Y, or X.Y.Z string, extract just the X / X.Y
518 # part and add Python to the front of it.
