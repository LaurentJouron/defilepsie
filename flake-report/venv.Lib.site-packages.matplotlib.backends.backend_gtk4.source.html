93 def _mpl_coords(self, xy=None):
94 """
95 Convert the *xy* position of a GTK event, or of the current cursor
96 position if *xy* is None, to Matplotlib coordinates.
97  
98 GTK use logical pixels, but the figure is scaled to physical pixels for
99 rendering. Transform to physical pixels so that all of the down-stream
100 transforms work as expected.
101  
102 Also, the origin is different and needs to be corrected.
103 """
104 if xy is None:
105 surface = self.get_native().get_surface()
106 is_over, x, y, mask = surface.get_device_position(
107 self.get_display().get_default_seat().get_pointer())
108 else:
109 x, y = xy
110 x = x * self.device_pixel_ratio
111 # flip y so y=0 is bottom of canvas
112 y = self.figure.bbox.height - y * self.device_pixel_ratio
113 return x, y

193 def _get_key(self, keyval, keycode, state):
194 unikey = chr(Gdk.keyval_to_unicode(keyval))
195 key = cbook._unikey_or_keysym_to_mplkey(
196 unikey,
197 Gdk.keyval_name(keyval))
198 modifiers = [
199 ("ctrl", Gdk.ModifierType.CONTROL_MASK, "control"),
200 ("alt", Gdk.ModifierType.ALT_MASK, "alt"),
201 ("shift", Gdk.ModifierType.SHIFT_MASK, "shift"),
202 ("super", Gdk.ModifierType.SUPER_MASK, "super"),
203 ]
204 mods = [
205 mod for mod, mask, mod_key in modifiers
206 if (mod_key != key and state & mask
207 and not (mod == "shift" and unikey.isprintable()))]
208 return "+".join([*mods, key])

506 def _normalize_shortcut(self, key):
507 """
508 Convert Matplotlib key presses to GTK+ accelerator identifiers.
509  
510 Related to `FigureCanvasGTK4._get_key`.
511 """
512 special = {
513 'backspace': 'BackSpace',
514 'pagedown': 'Page_Down',
515 'pageup': 'Page_Up',
516 'scroll_lock': 'Scroll_Lock',
517 }
518  
519 parts = key.split('+')
520 mods = ['<' + mod + '>' for mod in parts[:-1]]
521 key = parts[-1]
522  
523 if key in special:
524 key = special[key]
525 elif len(key) > 1:
526 key = key.capitalize()
527 elif key.isupper():
528 mods += ['<shift>']
529  
530 return ''.join(mods) + key

533 """
534 Check for a valid shortcut to be displayed.
535  
536 - GTK will never send 'cmd+' (see `FigureCanvasGTK4._get_key`).
537 - The shortcut window only shows keyboard shortcuts, not mouse buttons.
538 """
539 return 'cmd+' not in key and not key.startswith('MouseButton.')
