46 def relative_uri(base: str, to: str) -> str:
47 """Return a relative URL from ``base`` to ``to``."""
48 if to.startswith(SEP):
49 return to
50 b2 = base.split('#')[0].split(SEP)
51 t2 = to.split('#')[0].split(SEP)
52 # remove common segments (except the last segment)
53 for x, y in zip(b2[:-1], t2[:-1]):
54 if x != y:
55 break
56 b2.pop(0)
57 t2.pop(0)
58 if b2 == t2:
59 # Special case: relative_uri('f/index.html','f/index.html')
60 # returns '', not 'index.html'
61 return ''
62 if len(b2) == 1 and t2 == ['']:
63 # Special case: relative_uri('f/index.html','f/') should
64 # return './', not ''
65 return '.' + SEP
66 return ('..' + SEP) * (len(b2) - 1) + SEP.join(t2)
