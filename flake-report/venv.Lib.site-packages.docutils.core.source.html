218 def publish(self, argv=None, usage=None, description=None,
219 settings_spec=None, settings_overrides=None,
220 config_section=None, enable_exit_status=False):
221 """
222 Process command line options and arguments (if `self.settings` not
223 already set), run `self.reader` and then `self.writer`. Return
224 `self.writer`'s output.
225 """
226 exit = None
227 try:
228 if self.settings is None:
229 self.process_command_line(
230 argv, usage, description, settings_spec, config_section,
231 **(settings_overrides or {}))
232 self.set_io()
233 self.prompt()
234 self.document = self.reader.read(self.source, self.parser,
235 self.settings)
236 self.apply_transforms()
237 output = self.writer.write(self.document, self.destination)
238 self.writer.assemble_parts()
239 except SystemExit as error:
240 exit = True
241 exit_status = error.code
242 except Exception as error:
243 if not self.settings: # exception too early to report nicely
244 raise
245 if self.settings.traceback: # Propagate exceptions?
246 self.debugging_dumps()
247 raise
248 self.report_Exception(error)
249 exit = True
250 exit_status = 1
251 self.debugging_dumps()
252 if (enable_exit_status and self.document
253 and (self.document.reporter.max_level
254 >= self.settings.exit_status_level)):
255 sys.exit(self.document.reporter.max_level + 10)
256 elif exit:
257 sys.exit(exit_status)
258 return output

305 def report_Exception(self, error):
306 if isinstance(error, utils.SystemMessage):
307 self.report_SystemMessage(error)
308 elif isinstance(error, UnicodeEncodeError):
309 self.report_UnicodeError(error)
310 elif isinstance(error, io.InputError):
311 self._stderr.write('Unable to open source file for reading:\n'
312 ' %s\n' % io.error_string(error))
313 elif isinstance(error, io.OutputError):
314 self._stderr.write(
315 'Unable to open destination file for writing:\n'
316 ' %s\n' % io.error_string(error))
317 else:
318 print('%s' % io.error_string(error), file=self._stderr)
319 print(f"""\
320 Exiting due to error. Use "--traceback" to diagnose.
321 Please report errors to <docutils-users@lists.sourceforge.net>.
322 Include "--traceback" output, Docutils version ({__version__}\
323 {f' [{__version_details__}]' if __version_details__ else ''}),
324 Python version ({sys.version.split()[0]}), your OS type & version, \
325 and the command line used.""", file=self._stderr)

613 def publish_programmatically(source_class, source, source_path,
614 destination_class, destination, destination_path,
615 reader, reader_name,
616 parser, parser_name,
617 writer, writer_name,
618 settings, settings_spec,
619 settings_overrides, config_section,
620 enable_exit_status):
621 """
622 Set up & run a `Publisher` for custom programmatic use.
623  
624 Return the output (as `str` or `bytes`, depending on `destination_class`,
625 writer, and the "output_encoding" setting) and the Publisher object.
626  
627 Applications should not need to call this function directly. If it does
628 seem to be necessary to call this function directly, please write to the
629 Docutils-develop mailing list
630 <https://docutils.sourceforge.io/docs/user/mailing-lists.html#docutils-develop>.
631  
632 Parameters:
633  
634 * `source_class` **required**: The class for dynamically created source
635 objects. Typically `io.FileInput` or `io.StringInput`.
636  
637 * `source`: Type depends on `source_class`:
638  
639 - If `source_class` is `io.FileInput`: Either a file-like object
640 (must have 'read' and 'close' methods), or ``None``
641 (`source_path` is opened). If neither `source` nor
642 `source_path` are supplied, `sys.stdin` is used.
643  
644 - If `source_class` is `io.StringInput` **required**:
645 The input as either a `bytes` object (ensure the 'input_encoding'
646 setting matches its encoding) or a `str` object.
647  
648 * `source_path`: Type depends on `source_class`:
649  
650 - `io.FileInput`: Path to the input file, opened if no `source`
651 supplied.
652  
653 - `io.StringInput`: Optional. Path to the file or name of the
654 object that produced `source`. Only used for diagnostic output.
655  
656 * `destination_class` **required**: The class for dynamically created
657 destination objects. Typically `io.FileOutput` or `io.StringOutput`.
658  
659 * `destination`: Type depends on `destination_class`:
660  
661 - `io.FileOutput`: Either a file-like object (must have 'write' and
662 'close' methods), or ``None`` (`destination_path` is opened). If
663 neither `destination` nor `destination_path` are supplied,
664 `sys.stdout` is used.
665  
666 - `io.StringOutput`: Not used; pass ``None``.
667  
668 * `destination_path`: Type depends on `destination_class`:
669  
670 - `io.FileOutput`: Path to the output file. Opened if no `destination`
671 supplied.
672  
673 - `io.StringOutput`: Path to the file or object which will receive the
674 output; optional. Used for determining relative paths (stylesheets,
675 source links, etc.).
676  
677 * `reader`: A `docutils.readers.Reader` object.
678  
679 * `reader_name`: Name or alias of the Reader class to be instantiated if
680 no `reader` supplied.
681  
682 * `parser`: A `docutils.parsers.Parser` object.
683  
684 * `parser_name`: Name or alias of the Parser class to be instantiated if
685 no `parser` supplied.
686  
687 * `writer`: A `docutils.writers.Writer` object.
688  
689 * `writer_name`: Name or alias of the Writer class to be instantiated if
690 no `writer` supplied.
691  
692 * `settings`: A runtime settings (`docutils.frontend.Values`) object, for
693 dotted-attribute access to runtime settings. It's the end result of the
694 `SettingsSpec`, config file, and option processing. If `settings` is
695 passed, it's assumed to be complete and no further setting/config/option
696 processing is done.
697  
698 * `settings_spec`: A `docutils.SettingsSpec` subclass or object. Provides
699 extra application-specific settings definitions independently of
700 components. In other words, the application becomes a component, and
701 its settings data is processed along with that of the other components.
702 Used only if no `settings` specified.
703  
704 * `settings_overrides`: A dictionary containing application-specific
705 settings defaults that override the defaults of other components.
706 Used only if no `settings` specified.
707  
708 * `config_section`: A string, the name of the configuration file section
709 for this application. Overrides the ``config_section`` attribute
710 defined by `settings_spec`. Used only if no `settings` specified.
711  
712 * `enable_exit_status`: Boolean; enable exit status at end of processing?
713 """
714 publisher = Publisher(reader, parser, writer, settings=settings,
715 source_class=source_class,
716 destination_class=destination_class)
717 publisher.set_components(reader_name, parser_name, writer_name)
718 publisher.process_programmatic_settings(
719 settings_spec, settings_overrides, config_section)
720 publisher.set_source(source, source_path)
721 publisher.set_destination(destination, destination_path)
722 output = publisher.publish(enable_exit_status=enable_exit_status)
723 return output, publisher
