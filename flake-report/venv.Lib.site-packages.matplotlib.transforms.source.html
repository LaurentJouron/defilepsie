2 Matplotlib includes a framework for arbitrary geometric
3 transformations that is used determine the final position of all
4 elements drawn on the canvas.
5  
6 Transforms are composed into trees of `TransformNode` objects
7 whose actual value depends on their children. When the contents of
8 children change, their parents are automatically invalidated. The
9 next time an invalidated transform is accessed, it is recomputed to
10 reflect those changes. This invalidation/caching approach prevents
11 unnecessary recomputations of transforms, and contributes to better
12 interactive performance.
13  
14 For example, here is a graph of the transform tree used to plot data
15 to the graph:
16  
17 .. image:: ../_static/transforms.png
18  
19 The framework can be used for both affine and non-affine
20 transformations. However, for speed, we want to use the backend
21 renderers to perform affine transformations whenever possible.
22 Therefore, it is possible to perform just the affine or non-affine
23 part of a transformation on a set of data. The affine is always
24 assumed to occur after the non-affine. For any transform::
25  
26 full transform == non-affine part + affine part
27

1418 def __sub__(self, other):
1419 """
1420 Compose *self* with the inverse of *other*, cancelling identical terms
1421 if any::
1422  
1423 # In general:
1424 A - B == A + B.inverted()
1425 # (but see note regarding frozen transforms below).
1426  
1427 # If A "ends with" B (i.e. A == A' + B for some A') we can cancel
1428 # out B:
1429 (A' + B) - B == A'
1430  
1431 # Likewise, if B "starts with" A (B = A + B'), we can cancel out A:
1432 A - (A + B') == B'.inverted() == B'^-1
1433  
1434 Cancellation (rather than naively returning ``A + B.inverted()``) is
1435 important for multiple reasons:
1436  
1437 - It avoids floating-point inaccuracies when computing the inverse of
1438 B: ``B - B`` is guaranteed to cancel out exactly (resulting in the
1439 identity transform), whereas ``B + B.inverted()`` may differ by a
1440 small epsilon.
1441 - ``B.inverted()`` always returns a frozen transform: if one computes
1442 ``A + B + B.inverted()`` and later mutates ``B``, then
1443 ``B.inverted()`` won't be updated and the last two terms won't cancel
1444 out anymore; on the other hand, ``A + B - B`` will always be equal to
1445 ``A`` even if ``B`` is mutated.
1446 """
1447 # we only know how to do this operation if other is a Transform.
1448 if not isinstance(other, Transform):
1449 return NotImplemented
1450 for remainder, sub_tree in self._iter_break_from_left_to_right():
1451 if sub_tree == other:
1452 return remainder
1453 for remainder, sub_tree in other._iter_break_from_left_to_right():
1454 if sub_tree == self:
1455 if not remainder.has_inverse:
1456 raise ValueError(
1457 "The shortcut cannot be computed since 'other' "
1458 "includes a non-invertible component")
1459 return remainder.inverted()
1460 # if we have got this far, then there was no shortcut possible
1461 if other.has_inverse:
1462 return self + other.inverted()
1463 else:
1464 raise ValueError('It is not possible to compute transA - transB '
1465 'since transB cannot be inverted and there is no '
1466 'shortcut possible.')

1626

2817 def nonsingular(vmin, vmax, expander=0.001, tiny=1e-15, increasing=True):
2818 """
2819 Modify the endpoints of a range as needed to avoid singularities.
2820  
2821 Parameters
2822 ----------
2823 vmin, vmax : float
2824 The initial endpoints.
2825 expander : float, default: 0.001
2826 Fractional amount by which *vmin* and *vmax* are expanded if
2827 the original interval is too small, based on *tiny*.
2828 tiny : float, default: 1e-15
2829 Threshold for the ratio of the interval to the maximum absolute
2830 value of its endpoints. If the interval is smaller than
2831 this, it will be expanded. This value should be around
2832 1e-15 or larger; otherwise the interval will be approaching
2833 the double precision resolution limit.
2834 increasing : bool, default: True
2835 If True, swap *vmin*, *vmax* if *vmin* > *vmax*.
2836  
2837 Returns
2838 -------
2839 vmin, vmax : float
2840 Endpoints, expanded and/or swapped if necessary.
2841 If either input is inf or NaN, or if both inputs are 0 or very
2842 close to zero, it returns -*expander*, *expander*.
2843 """
2844  
2845 if (not np.isfinite(vmin)) or (not np.isfinite(vmax)):
2846 return -expander, expander
2847  
2848 swapped = False
2849 if vmax < vmin:
2850 vmin, vmax = vmax, vmin
2851 swapped = True
2852  
2853 # Expand vmin, vmax to float: if they were integer types, they can wrap
2854 # around in abs (abs(np.int8(-128)) == -128) and vmax - vmin can overflow.
2855 vmin, vmax = map(float, [vmin, vmax])
2856  
2857 maxabsvalue = max(abs(vmin), abs(vmax))
2858 if maxabsvalue < (1e6 / tiny) * np.finfo(float).tiny:
2859 vmin = -expander
2860 vmax = expander
2861  
2862 elif vmax - vmin <= maxabsvalue * tiny:
2863 if vmax == 0 and vmin == 0:
2864 vmin = -expander
2865 vmax = expander
2866 else:
2867 vmin -= expander*abs(vmin)
2868 vmax += expander*abs(vmax)
2869  
2870 if swapped and not increasing:
2871 vmin, vmax = vmax, vmin
2872 return vmin, vmax

2901
