72 def blit(photoimage, aggimage, offsets, bbox=None):
73 """
74 Blit *aggimage* to *photoimage*.
75  
76 *offsets* is a tuple describing how to fill the ``offset`` field of the
77 ``Tk_PhotoImageBlock`` struct: it should be (0, 1, 2, 3) for RGBA8888 data,
78 (2, 1, 0, 3) for little-endian ARBG32 (i.e. GBRA8888) data and (1, 2, 3, 0)
79 for big-endian ARGB32 (i.e. ARGB8888) data.
80  
81 If *bbox* is passed, it defines the region that gets blitted. That region
82 will be composed with the previous data according to the alpha channel.
83 Blitting will be clipped to pixels inside the canvas, including silently
84 doing nothing if the *bbox* region is entirely outside the canvas.
85  
86 Tcl events must be dispatched to trigger a blit from a non-Tcl thread.
87 """
88 data = np.asarray(aggimage)
89 height, width = data.shape[:2]
90 dataptr = (height, width, data.ctypes.data)
91 if bbox is not None:
92 (x1, y1), (x2, y2) = bbox.__array__()
93 x1 = max(math.floor(x1), 0)
94 x2 = min(math.ceil(x2), width)
95 y1 = max(math.floor(y1), 0)
96 y2 = min(math.ceil(y2), height)
97 if (x1 > x2) or (y1 > y2):
98 return
99 bboxptr = (x1, x2, y1, y2)
100 comp_rule = TK_PHOTO_COMPOSITE_OVERLAY
101 else:
102 bboxptr = (0, width, 0, height)
103 comp_rule = TK_PHOTO_COMPOSITE_SET
104  
105 # NOTE: _tkagg.blit is thread unsafe and will crash the process if called
106 # from a thread (GH#13293). Instead of blanking and blitting here,
107 # use tkapp.call to post a cross-thread event if this function is called
108 # from a non-Tcl thread.
109  
110 # tkapp.call coerces all arguments to strings, so to avoid string parsing
111 # within _blit, pack up the arguments into a global data structure.
112 args = photoimage, dataptr, offsets, bboxptr, comp_rule
113 # Need a unique key to avoid thread races.
114 # Again, make the key a string to avoid string parsing in _blit.
115 argsid = str(id(args))
116 _blit_args[argsid] = args

119 photoimage.tk.call(_blit_tcl_name, argsid)
120 except tk.TclError as e:
121 if "invalid command name" not in str(e):

144 def _on_timer(self):
145 super()._on_timer()
146 # Tk after() is only a single shot, so we need to add code here to
147 # reset the timer if we're not operating in single shot mode. However,
148 # if _timer is None, this means that _timer_stop has been called; so

361 def _mpl_modifiers(event, *, exclude=None):
362 # add modifier keys to the key string. Bit details originate from
363 # http://effbot.org/tkinterbook/tkinter-events-and-bindings.htm
364 # BIT_SHIFT = 0x001; BIT_CAPSLOCK = 0x002; BIT_CONTROL = 0x004;
365 # BIT_LEFT_ALT = 0x008; BIT_NUMLOCK = 0x010; BIT_RIGHT_ALT = 0x080;
366 # BIT_MB_1 = 0x100; BIT_MB_2 = 0x200; BIT_MB_3 = 0x400;
367 # In general, the modifier key is excluded from the modifier flag,
368 # however this is not the case on "darwin", so double check that
369 # we aren't adding repeat modifier flags to a modifier key.
370 modifiers = [
371 ("ctrl", 1 << 2, "control"),
372 ("alt", 1 << 17, "alt"),
373 ("shift", 1 << 0, "shift"),
374 ] if sys.platform == "win32" else [
375 ("ctrl", 1 << 2, "control"),
376 ("alt", 1 << 4, "alt"),
377 ("shift", 1 << 0, "shift"),
378 ("cmd", 1 << 3, "cmd"),
379 ] if sys.platform == "darwin" else [
380 ("ctrl", 1 << 2, "control"),
381 ("alt", 1 << 3, "alt"),
382 ("shift", 1 << 0, "shift"),
383 ("super", 1 << 6, "super"),
384 ]
385 return [name for name, mask, key in modifiers
386 if event.state & mask and exclude != key]
