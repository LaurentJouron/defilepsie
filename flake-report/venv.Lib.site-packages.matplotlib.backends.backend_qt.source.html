673 Construct a `.QIcon` from an image file *name*, including the extension
674 and relative to Matplotlib's "images" data directory.
675 """
676 # use a high-resolution icon with suffix '_large' if available
677 # note: user-provided icons may not have '_large' versions
678 path_regular = cbook._get_data_path('images', name)
679 path_large = path_regular.with_name(
680 path_regular.name.replace('.png', '_large.png'))
681 filename = str(path_large if path_large.exists() else path_regular)
682  
683 pm = QtGui.QPixmap(filename)
684 pm.setDevicePixelRatio(
685 self.devicePixelRatioF() or 1) # rarely, devicePixelRatioF=0
686 if self.palette().color(self.backgroundRole()).value() < 128:
687 icon_color = self.palette().color(self.foregroundRole())
688 mask = pm.createMaskFromColor(
689 QtGui.QColor('black'),
690 QtCore.Qt.MaskMode.MaskOutColor)
691 pm.fill(icon_color)
692 pm.setMask(mask)
693 return QtGui.QIcon(pm)

856 def _export_values(self):
857 # Explicitly round to 3 decimals (which is also the spinbox precision)
858 # to avoid numbers of the form 0.100...001.
859 self._export_values_dialog = QtWidgets.QDialog()
860 layout = QtWidgets.QVBoxLayout()
861 self._export_values_dialog.setLayout(layout)
862 text = QtWidgets.QPlainTextEdit()
863 text.setReadOnly(True)
864 layout.addWidget(text)
865 text.setPlainText(
866 ",\n".join(f"{attr}={spinbox.value():.3}"
867 for attr, spinbox in self._spinboxes.items()))
868 # Adjust the height of the text widget to fit the whole text, plus
869 # some padding.
870 size = text.maximumSize()
871 size.setHeight(
872 QtGui.QFontMetrics(text.document().defaultFont())
873 .size(0, text.toPlainText()).height() + 20)
874 text.setMaximumSize(size)
875 self._export_values_dialog.show()

877 def _on_value_changed(self):
878 spinboxes = self._spinboxes
879 # Set all mins and maxes, so that this can also be used in _reset().
880 for lower, higher in [("bottom", "top"), ("left", "right")]:
881 spinboxes[higher].setMinimum(spinboxes[lower].value() + .001)
882 spinboxes[lower].setMaximum(spinboxes[higher].value() - .001)
883 self._figure.subplots_adjust(
884 **{attr: spinbox.value() for attr, spinbox in spinboxes.items()})
885 self._figure.canvas.draw_idle()
886  
887 def _tight_layout(self):
888 self._figure.tight_layout()
889 for attr, spinbox in self._spinboxes.items():
890 spinbox.blockSignals(True)
891 spinbox.setValue(getattr(self._figure.subplotpars, attr))
892 spinbox.blockSignals(False)
893 self._figure.canvas.draw_idle()
894  
895 def _reset(self):
896 for spinbox, value in self._defaults.items():
897 spinbox.setRange(0, 1)
898 spinbox.blockSignals(True)
899 spinbox.setValue(value)
900 spinbox.blockSignals(False)
901 self._on_value_changed()
