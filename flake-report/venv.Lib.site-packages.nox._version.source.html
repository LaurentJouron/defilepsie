53 def _parse_needs_version(source: str, filename: str = "<unknown>") -> str | None:
54 """Parse ``nox.needs_version`` from the user's Noxfile."""
55 value: str | None = None
56 module: ast.Module = ast.parse(source, filename=filename)
57 for statement in module.body:
58 if isinstance(statement, ast.Assign):
59 for target in statement.targets:
60 if (
61 isinstance(target, ast.Attribute)
62 and isinstance(target.value, ast.Name)
63 and target.value.id == "nox"
64 and target.attr == "needs_version"
65 ):
66 value = _parse_string_constant(statement.value)
67 return value

78 def _check_nox_version_satisfies(needs_version: str) -> None:
79 """Check if the Nox version satisfies the given specifiers."""
80 version = Version(get_nox_version())

82 try:
83 specifiers = SpecifierSet(needs_version)
84 except InvalidSpecifier as error:
85 message = f"Cannot parse `nox.needs_version`: {error}"
86 with contextlib.suppress(InvalidVersion):
87 Version(needs_version)
88 message += f", did you mean '>= {needs_version}'?"
89 raise InvalidVersionSpecifier(message) from error

92 raise VersionCheckFailed(
