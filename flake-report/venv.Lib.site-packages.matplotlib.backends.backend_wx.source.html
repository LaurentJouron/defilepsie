1260 _instance = None # a reference to an open dialog singleton
1261 headers = [("Action", "Shortcuts", "Description")]
1262 widths = [100, 140, 300]
1263  
1264 def __init__(self, parent, help_entries):
1265 super().__init__(parent, title="Help",
1266 style=wx.DEFAULT_DIALOG_STYLE | wx.RESIZE_BORDER)
1267  
1268 sizer = wx.BoxSizer(wx.VERTICAL)
1269 grid_sizer = wx.FlexGridSizer(0, 3, 8, 6)
1270 # create and add the entries
1271 bold = self.GetFont().MakeBold()
1272 for r, row in enumerate(self.headers + help_entries):
1273 for (col, width) in zip(row, self.widths):
1274 label = wx.StaticText(self, label=col)
1275 if r == 0:
1276 label.SetFont(bold)
1277 label.Wrap(width)
1278 grid_sizer.Add(label, 0, 0, 0)
1279 # finalize layout, create button
1280 sizer.Add(grid_sizer, 0, wx.ALL, 6)
1281 ok = wx.Button(self, wx.ID_OK)
1282 sizer.Add(ok, 0, wx.ALIGN_CENTER_HORIZONTAL | wx.ALL, 8)
1283 self.SetSizer(sizer)
1284 sizer.Fit(self)
1285 self.Layout()
1286 self.Bind(wx.EVT_CLOSE, self._on_close)
1287 ok.Bind(wx.EVT_BUTTON, self._on_close)

1290 _HelpDialog._instance = None # remove global reference
1291 self.DestroyLater()
1292 event.Skip()
