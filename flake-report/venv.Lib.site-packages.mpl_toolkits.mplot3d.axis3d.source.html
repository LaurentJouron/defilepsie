16 def _move_from_center(coord, centers, deltas, axmask=(True, True, True)):

279 def _get_coord_info(self, renderer):
280 mins, maxs = np.array([
281 self.axes.get_xbound(),
282 self.axes.get_ybound(),
283 self.axes.get_zbound(),
284 ]).T
285  
286 # Get the mean value for each bound:
287 centers = 0.5 * (maxs + mins)
288  
289 # Add a small offset between min/max point and the edge of the plot:
290 deltas = (maxs - mins) / 12
291 mins -= 0.25 * deltas
292 maxs += 0.25 * deltas
293  
294 # Project the bounds along the current position of the cube:
295 bounds = mins[0], maxs[0], mins[1], maxs[1], mins[2], maxs[2]
296 bounds_proj = self.axes._tunit_cube(bounds, self.axes.M)
297  
298 # Determine which one of the parallel planes are higher up:
299 means_z0 = np.zeros(3)
300 means_z1 = np.zeros(3)
301 for i in range(3):
302 means_z0[i] = np.mean(bounds_proj[self._PLANES[2 * i], 2])
303 means_z1[i] = np.mean(bounds_proj[self._PLANES[2 * i + 1], 2])
304 highs = means_z0 < means_z1
305  
306 # Special handling for edge-on views
307 equals = np.abs(means_z0 - means_z1) <= np.finfo(float).eps
308 if np.sum(equals) == 2:
309 vertical = np.where(~equals)[0][0]
310 if vertical == 2: # looking at XY plane
311 highs = np.array([True, True, highs[2]])
312 elif vertical == 1: # looking at XZ plane
313 highs = np.array([True, highs[1], False])
314 elif vertical == 0: # looking at YZ plane
315 highs = np.array([highs[0], False, False])
316  
317 return mins, maxs, centers, deltas, bounds_proj, highs
