42 def get_dir_vector(zdir):
43 """
44 Return a direction vector.
45  
46 Parameters
47 ----------
48 zdir : {'x', 'y', 'z', None, 3-tuple}
49 The direction. Possible values are:
50  
51 - 'x': equivalent to (1, 0, 0)
52 - 'y': equivalent to (0, 1, 0)
53 - 'z': equivalent to (0, 0, 1)
54 - *None*: equivalent to (0, 0, 0)
55 - an iterable (x, y, z) is converted to an array
56  
57 Returns
58 -------
59 x, y, z : array
60 The direction vector.
61 """
62 if zdir == 'x':
63 return np.array((1, 0, 0))
64 elif zdir == 'y':
65 return np.array((0, 1, 0))
66 elif zdir == 'z':
67 return np.array((0, 0, 1))
68 elif zdir is None:
69 return np.array((0, 0, 0))
70 elif np.iterable(zdir) and len(zdir) == 3:
71 return np.array(zdir)
72 else:
73 raise ValueError("'x', 'y', 'z', None or vector of length 3 expected")

844 class Poly3DCollection(PolyCollection):
845 """
846 A collection of 3D polygons.
847  
848 .. note::
849 **Filling of 3D polygons**
850  
851 There is no simple definition of the enclosed surface of a 3D polygon
852 unless the polygon is planar.
853  
854 In practice, Matplotlib fills the 2D projection of the polygon. This
855 gives a correct filling appearance only for planar polygons. For all
856 other polygons, you'll find orientations in which the edges of the
857 polygon intersect in the projection. This will lead to an incorrect
858 visualization of the 3D area.
859  
860 If you need filled areas, it is recommended to create them via
861 `~mpl_toolkits.mplot3d.axes3d.Axes3D.plot_trisurf`, which creates a
862 triangulation and thus generates consistent surfaces.

998 def do_3d_projection(self):
999 """
1000 Perform the 3D projection for this object.
1001 """
1002 if self._A is not None:
1003 # force update of color mapping because we re-order them
1004 # below. If we do not do this here, the 2D draw will call
1005 # this, but we will never port the color mapped values back
1006 # to the 3D versions.
1007 #
1008 # We hold the 3D versions in a fixed order (the order the user
1009 # passed in) and sort the 2D version by view depth.
1010 self.update_scalarmappable()
1011 if self._face_is_mapped:
1012 self._facecolor3d = self._facecolors
1013 if self._edge_is_mapped:
1014 self._edgecolor3d = self._edgecolors
1015 txs, tys, tzs = proj3d._proj_transform_vec(self._vec, self.axes.M)
1016 xyzlist = [(txs[sl], tys[sl], tzs[sl]) for sl in self._segslices]
1017  
1018 # This extra fuss is to re-order face / edge colors
1019 cface = self._facecolor3d
1020 cedge = self._edgecolor3d
1021 if len(cface) != len(xyzlist):
1022 cface = cface.repeat(len(xyzlist), axis=0)
1023 if len(cedge) != len(xyzlist):
1024 if len(cedge) == 0:
1025 cedge = cface
1026 else:
1027 cedge = cedge.repeat(len(xyzlist), axis=0)
1028  
1029 if xyzlist:
1030 # sort by depth (furthest drawn first)
1031 z_segments_2d = sorted(
1032 ((self._zsortfunc(zs), np.column_stack([xs, ys]), fc, ec, idx)
1033 for idx, ((xs, ys, zs), fc, ec)
1034 in enumerate(zip(xyzlist, cface, cedge))),
1035 key=lambda x: x[0], reverse=True)
1036  
1037 _, segments_2d, self._facecolors2d, self._edgecolors2d, idxs = \
1038 zip(*z_segments_2d)
1039 else:
1040 segments_2d = []
1041 self._facecolors2d = np.empty((0, 4))
1042 self._edgecolors2d = np.empty((0, 4))
1043 idxs = []
1044  
1045 if self._codes3d is not None:
1046 codes = [self._codes3d[idx] for idx in idxs]
1047 PolyCollection.set_verts_and_codes(self, segments_2d, codes)
1048 else:
1049 PolyCollection.set_verts(self, segments_2d, self._closed)
1050  
1051 if len(self._edgecolor3d) != len(cface):
1052 self._edgecolors2d = self._edgecolor3d
1053  
1054 # Return zorder value
1055 if self._sort_zpos is not None:
1056 zvec = np.array([[0], [0], [self._sort_zpos], [1]])
1057 ztrans = proj3d._proj_transform_vec(zvec, self.axes.M)
1058 return ztrans[2][0]
1059 elif tzs.size > 0:
1060 # FIXME: Some results still don't look quite right.
1061 # In particular, examine contourf3d_demo2.py
1062 # with az = -54 and elev = -45.
1063 return np.min(tzs)
1064 else:
1065 return np.nan

1160 def _zalpha(colors, zs):
1161 """Modify the alphas of the color list according to depth."""
1162 # FIXME: This only works well if the points for *zs* are well-spaced
1163 # in all three dimensions. Otherwise, at certain orientations,
1164 # the min and max zs are very close together.
1165 # Should really normalize against the viewing depth.
1166 if len(colors) == 0 or len(zs) == 0:
1167 return np.zeros((0, 4))
1168 norm = Normalize(min(zs), max(zs))
1169 sats = 1 - norm(zs) * 0.7
1170 rgba = np.broadcast_to(mcolors.to_rgba_array(colors), (len(zs), 4))
1171 return np.column_stack([rgba[:, :3], rgba[:, 3] * sats])

1174 def _generate_normals(polygons):
1175 """
1176 Compute the normals of a list of polygons, one normal per polygon.
1177  
1178 Normals point towards the viewer for a face with its vertices in
1179 counterclockwise order, following the right hand rule.
1180  
1181 Uses three points equally spaced around the polygon. This method assumes
1182 that the points are in a plane. Otherwise, more than one shade is required,
1183 which is not supported.
1184  
1185 Parameters
1186 ----------
1187 polygons : list of (M_i, 3) array-like, or (..., M, 3) array-like
1188 A sequence of polygons to compute normals for, which can have
1189 varying numbers of vertices. If the polygons all have the same
1190 number of vertices and array is passed, then the operation will
1191 be vectorized.
1192  
1193 Returns
1194 -------
1195 normals : (..., 3) array
1196 A normal vector estimated for the polygon.
1197 """
1198 if isinstance(polygons, np.ndarray):
1199 # optimization: polygons all have the same number of points, so can
1200 # vectorize
1201 n = polygons.shape[-2]
1202 i1, i2, i3 = 0, n//3, 2*n//3
1203 v1 = polygons[..., i1, :] - polygons[..., i2, :]
1204 v2 = polygons[..., i2, :] - polygons[..., i3, :]
1205 else:
1206 # The subtraction doesn't vectorize because polygons is jagged.
1207 v1 = np.empty((len(polygons), 3))
1208 v2 = np.empty((len(polygons), 3))
1209 for poly_i, ps in enumerate(polygons):
1210 n = len(ps)
1211 i1, i2, i3 = 0, n//3, 2*n//3
1212 v1[poly_i, :] = ps[i1, :] - ps[i2, :]
1213 v2[poly_i, :] = ps[i2, :] - ps[i3, :]
1214 return np.cross(v1, v2)

1217 def _shade_colors(color, normals, lightsource=None):
1218 """
1219 Shade *color* using normal vectors given by *normals*,
1220 assuming a *lightsource* (using default position if not given).
1221 *color* can also be an array of the same length as *normals*.
1222 """
1223 if lightsource is None:
1224 # chosen for backwards-compatibility
1225 lightsource = mcolors.LightSource(azdeg=225, altdeg=19.4712)
1226  
1227 with np.errstate(invalid="ignore"):
1228 shade = ((normals / np.linalg.norm(normals, axis=1, keepdims=True))
1229 @ lightsource.direction)
1230 mask = ~np.isnan(shade)
1231  
1232 if mask.any():
1233 # convert dot product to allowed shading fractions
1234 in_norm = mcolors.Normalize(-1, 1)
1235 out_norm = mcolors.Normalize(0.3, 1).inverse
1236  
1237 def norm(x):
1238 return out_norm(in_norm(x))
1239  
1240 shade[~mask] = 0
1241  
1242 color = mcolors.to_rgba_array(color)
1243 # shape of color should be (M, 4) (where M is number of faces)
1244 # shape of shade should be (M,)
1245 # colors should have final shape of (M, 4)
1246 alpha = color[:, 3]
1247 colors = norm(shade)[:, np.newaxis] * color
1248 colors[:, 3] = alpha
1249 else:
1250 colors = np.asanyarray(color).copy()
1251  
1252 return colors
