113 def setup_logging(
114 color: bool, verbose: bool = False, add_timestamp: bool = False
115 ) -> None: # pragma: no cover
116 """Setup logging.
117  
118 Args:
119 color (bool): If true, the output will be colored using
120 colorlog. Otherwise, it will be plaintext.
121 """
122 root_logger = logging.getLogger()
123 if verbose:
124 root_logger.setLevel(OUTPUT)
125 else:
126 root_logger.setLevel(logging.DEBUG)
127 handler = logging.StreamHandler()
128  
129 handler.setFormatter(_get_formatter(color, add_timestamp))
130 root_logger.addHandler(handler)
