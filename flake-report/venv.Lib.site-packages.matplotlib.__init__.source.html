202 def _parse_to_version_info(version_str):
203 """
204 Parse a version string to a namedtuple analogous to sys.version_info.
205  
206 See:
207 https://packaging.pypa.io/en/latest/version.html#packaging.version.parse
208 https://docs.python.org/3/library/sys.html#sys.version_info
209 """
210 v = parse_version(version_str)
211 if v.pre is None and v.post is None and v.dev is None:
212 return _VersionInfo(v.major, v.minor, v.micro, 'final', 0)
213 elif v.dev is not None:
214 return _VersionInfo(v.major, v.minor, v.micro, 'alpha', v.dev)
215 elif v.pre is not None:
216 releaselevel = {
217 'a': 'alpha',
218 'b': 'beta',
219 'rc': 'candidate'}.get(v.pre[0], 'alpha')
220 return _VersionInfo(v.major, v.minor, v.micro, releaselevel, v.pre[1])
221 else:
222 # fallback for v.post: guess-next-dev scheme from setuptools_scm
223 return _VersionInfo(v.major, v.minor, v.micro + 1, 'alpha', v.post)

326 """

601 def matplotlib_fname():
602 """
603 Get the location of the config file.
604  
605 The file location is determined in the following order
606  
607 - ``$PWD/matplotlibrc``
608 - ``$MATPLOTLIBRC`` if it is not a directory
609 - ``$MATPLOTLIBRC/matplotlibrc``
610 - ``$MPLCONFIGDIR/matplotlibrc``
611 - On Linux,
612 - ``$XDG_CONFIG_HOME/matplotlib/matplotlibrc`` (if ``$XDG_CONFIG_HOME``
613 is defined)
614 - or ``$HOME/.config/matplotlib/matplotlibrc`` (if ``$XDG_CONFIG_HOME``
615 is not defined)
616 - On other platforms,
617 - ``$HOME/.matplotlib/matplotlibrc`` if ``$HOME`` is defined
618 - Lastly, it looks in ``$MATPLOTLIBDATA/matplotlibrc``, which should always
619 exist.
620 """
621  
622 def gen_candidates():
623 # rely on down-stream code to make absolute. This protects us
624 # from having to directly get the current working directory
625 # which can fail if the user has ended up with a cwd that is
626 # non-existent.
627 yield 'matplotlibrc'
628 try:
629 matplotlibrc = os.environ['MATPLOTLIBRC']
630 except KeyError:
631 pass
632 else:
633 yield matplotlibrc
634 yield os.path.join(matplotlibrc, 'matplotlibrc')
635 yield os.path.join(get_configdir(), 'matplotlibrc')
636 yield os.path.join(get_data_path(), 'matplotlibrc')
637  
638 for fname in gen_candidates():
639 if os.path.exists(fname) and not os.path.isdir(fname):
640 return fname

860 def _rc_params_in_file(fname, transform=lambda x: x, fail_on_error=False):
861 """
862 Construct a `RcParams` instance from file *fname*.
863  
864 Unlike `rc_params_from_file`, the configuration class only contains the
865 parameters specified in the file (i.e. default values are not filled in).
866  
867 Parameters
868 ----------
869 fname : path-like
870 The loaded file.
871 transform : callable, default: the identity function
872 A function called on each individual line of the file to transform it,
873 before further parsing.
874 fail_on_error : bool, default: False
875 Whether invalid entries should result in an exception or a warning.
876 """
877 import matplotlib as mpl
878 rc_temp = {}
879 with _open_file_or_url(fname) as fd:
880 try:
881 for line_no, line in enumerate(fd, 1):
882 line = transform(line)
883 strippedline = cbook._strip_comment(line)
884 if not strippedline:
885 continue
886 tup = strippedline.split(':', 1)
887 if len(tup) != 2:
888 _log.warning('Missing colon in file %r, line %d (%r)',
889 fname, line_no, line.rstrip('\n'))
890 continue
891 key, val = tup
892 key = key.strip()
893 val = val.strip()
894 if val.startswith('"') and val.endswith('"'):
895 val = val[1:-1] # strip double quotes
896 if key in rc_temp:
897 _log.warning('Duplicate key in file %r, line %d (%r)',
898 fname, line_no, line.rstrip('\n'))
899 rc_temp[key] = (val, line, line_no)
900 except UnicodeDecodeError:
901 _log.warning('Cannot decode configuration file %r as utf-8.',
902 fname)
903 raise
904  
905 config = RcParams()
906  
907 for key, (val, line, line_no) in rc_temp.items():
908 if key in rcsetup._validators:
909 if fail_on_error:
910 config[key] = val # try to convert to proper type or raise
911 else:
912 try:
913 config[key] = val # try to convert to proper type or skip
914 except Exception as msg:
915 _log.warning('Bad value in file %r, line %d (%r): %s',
916 fname, line_no, line.rstrip('\n'), msg)
917 elif key in _deprecated_ignore_map:
918 version, alt_key = _deprecated_ignore_map[key]
919 _api.warn_deprecated(
920 version, name=key, alternative=alt_key, obj_type='rcparam',
921 addendum="Please update your matplotlibrc.")
922 else:
923 # __version__ must be looked up as an attribute to trigger the
924 # module-level __getattr__.
925 version = ('main' if '.post' in mpl.__version__
926 else f'v{mpl.__version__}')
927 _log.warning("""
928 Bad key %(key)s in file %(fname)s, line %(line_no)s (%(line)r)
929 You probably need to get an updated matplotlibrc file from
930 https://github.com/matplotlib/matplotlib/blob/%(version)s/lib/matplotlib/mpl-data/matplotlibrc
931 or from the matplotlib source distribution""",
932 dict(key=key, fname=fname, line_no=line_no,
933 line=line.rstrip('\n'), version=version))
934 return config

1005 def rc(group, **kwargs):
1006 """
1007 Set the current `.rcParams`. *group* is the grouping for the rc, e.g.,
1008 for ``lines.linewidth`` the group is ``lines``, for
1009 ``axes.facecolor``, the group is ``axes``, and so on. Group may
1010 also be a list or tuple of group names, e.g., (*xtick*, *ytick*).
1011 *kwargs* is a dictionary attribute name/value pairs, e.g.,::
1012  
1013 rc('lines', linewidth=2, color='r')
1014  
1015 sets the current `.rcParams` and is equivalent to::
1016  
1017 rcParams['lines.linewidth'] = 2
1018 rcParams['lines.color'] = 'r'
1019  
1020 The following aliases are available to save typing for interactive users:
1021  
1022 ===== =================
1023 Alias Property
1024 ===== =================
1025 'lw' 'linewidth'
1026 'ls' 'linestyle'
1027 'c' 'color'
1028 'fc' 'facecolor'
1029 'ec' 'edgecolor'
1030 'mew' 'markeredgewidth'
1031 'aa' 'antialiased'
1032 ===== =================
1033  
1034 Thus you could abbreviate the above call as::
1035  
1036 rc('lines', lw=2, c='r')
1037  
1038 Note you can use python's kwargs dictionary facility to store
1039 dictionaries of default parameters. e.g., you can customize the
1040 font rc as follows::
1041  
1042 font = {'family' : 'monospace',
1043 'weight' : 'bold',
1044 'size' : 'larger'}
1045 rc('font', **font) # pass in the font dict as kwargs
1046  
1047 This enables you to easily switch between several configurations. Use
1048 ``matplotlib.style.use('default')`` or :func:`~matplotlib.rcdefaults` to
1049 restore the default `.rcParams` after changes.
1050  
1051 Notes
1052 -----
1053 Similar functionality is available by using the normal dict interface, i.e.
1054 ``rcParams.update({"lines.linewidth": 2, ...})`` (but ``rcParams.update``
1055 does not support abbreviations or grouping).
1056 """
1057  
1058 aliases = {
1059 'lw': 'linewidth',
1060 'ls': 'linestyle',
1061 'c': 'color',
1062 'fc': 'facecolor',
1063 'ec': 'edgecolor',
1064 'mew': 'markeredgewidth',
1065 'aa': 'antialiased',
1066 }
1067  
1068 if isinstance(group, str):
1069 group = (group,)
1070 for g in group:
1071 for k, v in kwargs.items():
1072 name = aliases.get(k) or k
1073 key = f'{g}.{name}'
1074 try:
1075 rcParams[key] = v
1076 except KeyError as err:
1077 raise KeyError(('Unrecognized key "%s" for group "%s" and '
1078 'name "%s"') % (key, g, name)) from err

1159 ----------

1168 :ref:`customizing-with-matplotlibrc-files`

1171 --------

1205 def use(backend, *, force=True):
1206 """
1207 Select the backend used for rendering and GUI integration.
1208  
1209 If pyplot is already imported, `~matplotlib.pyplot.switch_backend` is used
1210 and if the new backend is different than the current backend, all Figures
1211 will be closed.
1212  
1213 Parameters
1214 ----------
1215 backend : str
1216 The backend to switch to. This can either be one of the standard
1217 backend names, which are case-insensitive:
1218  
1219 - interactive backends:
1220 GTK3Agg, GTK3Cairo, GTK4Agg, GTK4Cairo, MacOSX, nbAgg, QtAgg,
1221 QtCairo, TkAgg, TkCairo, WebAgg, WX, WXAgg, WXCairo, Qt5Agg, Qt5Cairo
1222  
1223 - non-interactive backends:
1224 agg, cairo, pdf, pgf, ps, svg, template
1225  
1226 or a string of the form: ``module://my.module.name``.
1227  
1228 Switching to an interactive backend is not possible if an unrelated
1229 event loop has already been started (e.g., switching to GTK3Agg if a
1230 TkAgg window has already been opened). Switching to a non-interactive
1231 backend is always possible.
1232  
1233 force : bool, default: True
1234 If True (the default), raise an `ImportError` if the backend cannot be
1235 set up (either because it fails to import, or because an incompatible
1236 GUI interactive framework is already running); if False, silently
1237 ignore the failure.
1238  
1239 See Also
1240 --------
1241 :ref:`backends`
1242 matplotlib.get_backend
1243 matplotlib.pyplot.switch_backend
1244  
1245 """
1246 name = validate_backend(backend)
1247 # don't (prematurely) resolve the "auto" backend setting
1248 if rcParams._get_backend_or_none() == name:
1249 # Nothing to do if the requested backend is already set
1250 pass
1251 else:
1252 # if pyplot is not already imported, do not import it. Doing
1253 # so may trigger a `plt.switch_backend` to the _default_ backend
1254 # before we get a chance to change to the one the user just requested
1255 plt = sys.modules.get('matplotlib.pyplot')
1256 # if pyplot is imported, then try to change backends
1257 if plt is not None:
1258 try:
1259 # we need this import check here to re-raise if the
1260 # user does not have the libraries to support their
1261 # chosen backend installed.
1262 plt.switch_backend(name)
1263 except ImportError:
1264 if force:
1265 raise
1266 # if we have not imported pyplot, then we can set the rcParam
1267 # value which will be respected when the user finally imports
1268 # pyplot
1269 else:
1270 rcParams['backend'] = backend
1271 # if the user has asked for a given backend, do not helpfully
1272 # fallback
1273 rcParams['backend_fallback'] = False
1274  
1275  
1276 if os.environ.get('MPLBACKEND'):
1277 rcParams['backend'] = os.environ.get('MPLBACKEND')

1335 def _replacer(data, value):
1336 """
1337 Either returns ``data[value]`` or passes ``data`` back, converts either to
1338 a sequence.
1339 """
1340 try:
1341 # if key isn't a string don't bother
1342 if isinstance(value, str):
1343 # try to use __getitem__
1344 value = data[value]
1345 except Exception:
1346 # key does not exist, silently fall back to key
1347 pass
1348 return sanitize_sequence(value)

1404 def _preprocess_data(func=None, *, replace_names=None, label_namer=None):
1405 """
1406 A decorator to add a 'data' kwarg to a function.
1407  
1408 When applied::
1409  
1410 @_preprocess_data()
1411 def func(ax, *args, **kwargs): ...
1412  
1413 the signature is modified to ``decorated(ax, *args, data=None, **kwargs)``
1414 with the following behavior:
1415  
1416 - if called with ``data=None``, forward the other arguments to ``func``;
1417 - otherwise, *data* must be a mapping; for any argument passed in as a
1418 string ``name``, replace the argument by ``data[name]`` (if this does not
1419 throw an exception), then forward the arguments to ``func``.
1420  
1421 In either case, any argument that is a `MappingView` is also converted to a
1422 list.
1423  
1424 Parameters
1425 ----------
1426 replace_names : list of str or None, default: None
1427 The list of parameter names for which lookup into *data* should be
1428 attempted. If None, replacement is attempted for all arguments.
1429 label_namer : str, default: None
1430 If set e.g. to "namer" (which must be a kwarg in the function's
1431 signature -- not as ``**kwargs``), if the *namer* argument passed in is
1432 a (string) key of *data* and no *label* kwarg is passed, then use the
1433 (string) value of the *namer* as *label*. ::
1434  
1435 @_preprocess_data(label_namer="foo")
1436 def func(foo, label=None): ...
1437  
1438 func("key", data={"key": value})
1439 # is equivalent to
1440 func.__wrapped__(value, label="key")
1441 """
1442  
1443 if func is None: # Return the actual decorator.
1444 return functools.partial(
1445 _preprocess_data,
1446 replace_names=replace_names, label_namer=label_namer)
1447  
1448 sig = inspect.signature(func)
1449 varargs_name = None
1450 varkwargs_name = None
1451 arg_names = []
1452 params = list(sig.parameters.values())
1453 for p in params:
1454 if p.kind is Parameter.VAR_POSITIONAL:
1455 varargs_name = p.name
1456 elif p.kind is Parameter.VAR_KEYWORD:
1457 varkwargs_name = p.name
1458 else:
1459 arg_names.append(p.name)
1460 data_param = Parameter("data", Parameter.KEYWORD_ONLY, default=None)
1461 if varkwargs_name:
1462 params.insert(-1, data_param)
1463 else:
1464 params.append(data_param)
1465 new_sig = sig.replace(parameters=params)
1466 arg_names = arg_names[1:] # remove the first "ax" / self arg
1467  
1468 assert {*arg_names}.issuperset(replace_names or []) or varkwargs_name, (
1469 "Matplotlib internal error: invalid replace_names "
1470 f"({replace_names!r}) for {func.__name__!r}")
1471 assert label_namer is None or label_namer in arg_names, (
1472 "Matplotlib internal error: invalid label_namer "
1473 f"({label_namer!r}) for {func.__name__!r}")
1474  
1475 @functools.wraps(func)
1476 def inner(ax, *args, data=None, **kwargs):
1477 if data is None:
1478 return func(ax, *map(sanitize_sequence, args), **kwargs)
1479  
1480 bound = new_sig.bind(ax, *args, **kwargs)
1481 auto_label = (bound.arguments.get(label_namer)
1482 or bound.kwargs.get(label_namer))
1483  
1484 for k, v in bound.arguments.items():
1485 if k == varkwargs_name:
1486 for k1, v1 in v.items():
1487 if replace_names is None or k1 in replace_names:
1488 v[k1] = _replacer(data, v1)
1489 elif k == varargs_name:
1490 if replace_names is None:
1491 bound.arguments[k] = tuple(_replacer(data, v1) for v1 in v)
1492 else:
1493 if replace_names is None or k in replace_names:
1494 bound.arguments[k] = _replacer(data, v)
1495  
1496 new_args = bound.args
1497 new_kwargs = bound.kwargs
1498  
1499 args_and_kwargs = {**bound.arguments, **bound.kwargs}
1500 if label_namer and "label" not in args_and_kwargs:
1501 new_kwargs["label"] = _label_from_arg(
1502 args_and_kwargs.get(label_namer), auto_label)
1503  
1504 return func(*new_args, **new_kwargs)
1505  
1506 inner.__doc__ = _add_data_doc(inner.__doc__, replace_names)
1507 inner.__signature__ = new_sig
1508 return inner
