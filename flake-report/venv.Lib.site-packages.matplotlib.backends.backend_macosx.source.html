192 def _maybe_allow_interrupt():
193 """
194 This manager allows to terminate a plot by sending a SIGINT. It is
195 necessary because the running backend prevents Python interpreter to
196 run and process signals (i.e., to raise KeyboardInterrupt exception). To
197 solve this one needs to somehow wake up the interpreter and make it close
198 the plot window. The implementation is taken from qt_compat, see that
199 docstring for a more detailed description.
200 """
201 old_sigint_handler = signal.getsignal(signal.SIGINT)
202 if old_sigint_handler in (None, signal.SIG_IGN, signal.SIG_DFL):
203 yield
204 return
205  
206 handler_args = None
207 wsock, rsock = socket.socketpair()
208 wsock.setblocking(False)
209 rsock.setblocking(False)
210 old_wakeup_fd = signal.set_wakeup_fd(wsock.fileno())
211 _macosx.wake_on_fd_write(rsock.fileno())

213 def handle(*args):
214 nonlocal handler_args
215 handler_args = args
216 _macosx.stop()
217  
218 signal.signal(signal.SIGINT, handle)
219 try:
220 yield
221 finally:
222 wsock.close()
223 rsock.close()
