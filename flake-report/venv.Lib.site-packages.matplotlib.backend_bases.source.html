229 def draw_path_collection(self, gc, master_transform, paths, all_transforms,
230 offsets, offset_trans, facecolors, edgecolors,
231 linewidths, linestyles, antialiaseds, urls,
232 offset_position):
233 """
234 Draw a collection of *paths*.
235  
236 Each path is first transformed by the corresponding entry
237 in *all_transforms* (a list of (3, 3) matrices) and then by
238 *master_transform*. They are then translated by the corresponding
239 entry in *offsets*, which has been first transformed by *offset_trans*.
240  
241 *facecolors*, *edgecolors*, *linewidths*, *linestyles*, and
242 *antialiased* are lists that set the corresponding properties.
243  
244 *offset_position* is unused now, but the argument is kept for
245 backwards compatibility.
246  
247 The base (fallback) implementation makes multiple calls to `draw_path`.
248 Backends may want to override this in order to render each set of
249 path data only once, and then reference that path multiple times with
250 the different offsets, colors, styles etc. The generator methods
251 `_iter_collection_raw_paths` and `_iter_collection` are provided to
252 help with (and standardize) the implementation across backends. It
253 is highly recommended to use those generators, so that changes to the
254 behavior of `draw_path_collection` can be made globally.
255 """
256 path_ids = self._iter_collection_raw_paths(master_transform,
257 paths, all_transforms)
258  
259 for xo, yo, path_id, gc0, rgbFace in self._iter_collection(
260 gc, list(path_ids), offsets, offset_trans,
261 facecolors, edgecolors, linewidths, linestyles,
262 antialiaseds, urls, offset_position):
263 path, transform = path_id
264 # Only apply another translation if we have an offset, else we
265 # reuse the initial transform.
266 if xo != 0 or yo != 0:
267 # The transformation can be used by multiple paths. Since
268 # translate is a inplace operation, we need to copy the
269 # transformation by .frozen() before applying the translation.
270 transform = transform.frozen()
271 transform.translate(xo, yo)
272 self.draw_path(gc0, path, transform, rgbFace)

331 def _iter_collection_raw_paths(self, master_transform, paths,
332 all_transforms):
333 """
334 Helper method (along with `_iter_collection`) to implement
335 `draw_path_collection` in a memory-efficient manner.
336  
337 This method yields all of the base path/transform combinations, given a
338 master transform, a list of paths and list of transforms.
339  
340 The arguments should be exactly what is passed in to
341 `draw_path_collection`.
342  
343 The backend should take each yielded path and transform and create an
344 object that can be referenced (reused) later.
345 """
346 Npaths = len(paths)
347 Ntransforms = len(all_transforms)
348 N = max(Npaths, Ntransforms)
349  
350 if Npaths == 0:
351 return
352  
353 transform = transforms.IdentityTransform()
354 for i in range(N):
355 path = paths[i % Npaths]
356 if Ntransforms:
357 transform = Affine2D(all_transforms[i % Ntransforms])
358 yield path, transform + master_transform

360 def _iter_collection_uses_per_path(self, paths, all_transforms,
361 offsets, facecolors, edgecolors):
362 """
363 Compute how many times each raw path object returned by
364 `_iter_collection_raw_paths` would be used when calling
365 `_iter_collection`. This is intended for the backend to decide
366 on the tradeoff between using the paths in-line and storing
367 them once and reusing. Rounds up in case the number of uses
368 is not the same for every path.
369 """
370 Npaths = len(paths)
371 if Npaths == 0 or len(facecolors) == len(edgecolors) == 0:
372 return 0
373 Npath_ids = max(Npaths, len(all_transforms))
374 N = max(Npath_ids, len(offsets))
375 return (N + Npath_ids - 1) // Npath_ids

377 def _iter_collection(self, gc, path_ids, offsets, offset_trans, facecolors,
378 edgecolors, linewidths, linestyles,
379 antialiaseds, urls, offset_position):
380 """
381 Helper method (along with `_iter_collection_raw_paths`) to implement
382 `draw_path_collection` in a memory-efficient manner.
383  
384 This method yields all of the path, offset and graphics context
385 combinations to draw the path collection. The caller should already
386 have looped over the results of `_iter_collection_raw_paths` to draw
387 this collection.
388  
389 The arguments should be the same as that passed into
390 `draw_path_collection`, with the exception of *path_ids*, which is a
391 list of arbitrary objects that the backend will use to reference one of
392 the paths created in the `_iter_collection_raw_paths` stage.
393  
394 Each yielded result is of the form::
395  
396 xo, yo, path_id, gc, rgbFace
397  
398 where *xo*, *yo* is an offset; *path_id* is one of the elements of
399 *path_ids*; *gc* is a graphics context and *rgbFace* is a color to
400 use for filling the path.
401 """
402 Npaths = len(path_ids)
403 Noffsets = len(offsets)
404 N = max(Npaths, Noffsets)
405 Nfacecolors = len(facecolors)
406 Nedgecolors = len(edgecolors)
407 Nlinewidths = len(linewidths)
408 Nlinestyles = len(linestyles)
409 Nurls = len(urls)
410  
411 if (Nfacecolors == 0 and Nedgecolors == 0) or Npaths == 0:
412 return
413  
414 gc0 = self.new_gc()
415 gc0.copy_properties(gc)

417 def cycle_or_default(seq, default=None):
418 # Cycle over *seq* if it is not empty; else always yield *default*.
419 return (itertools.cycle(seq) if len(seq)
420 else itertools.repeat(default))
421  
422 pathids = cycle_or_default(path_ids)
423 toffsets = cycle_or_default(offset_trans.transform(offsets), (0, 0))
424 fcs = cycle_or_default(facecolors)
425 ecs = cycle_or_default(edgecolors)
426 lws = cycle_or_default(linewidths)
427 lss = cycle_or_default(linestyles)
428 aas = cycle_or_default(antialiaseds)
429 urls = cycle_or_default(urls)
430  
431 if Nedgecolors == 0:
432 gc0.set_linewidth(0.0)
433  
434 for pathid, (xo, yo), fc, ec, lw, ls, aa, url in itertools.islice(
435 zip(pathids, toffsets, fcs, ecs, lws, lss, aas, urls), N):
436 if not (np.isfinite(xo) and np.isfinite(yo)):
437 continue
438 if Nedgecolors:
439 if Nlinewidths:
440 gc0.set_linewidth(lw)
441 if Nlinestyles:
442 gc0.set_dashes(*ls)
443 if len(ec) == 4 and ec[3] == 0.0:
444 gc0.set_linewidth(0)
445 else:
446 gc0.set_foreground(ec)
447 if fc is not None and len(fc) == 4 and fc[3] == 0:
448 fc = None
449 gc0.set_antialiased(aa)
450 if Nurls:
451 gc0.set_url(url)
452 yield xo, yo, pathid, gc0, fc
453 gc0.restore()

574 def _get_text_path_transform(self, x, y, s, prop, angle, ismath):
575 """
576 Return the text path and transform.
577  
578 Parameters
579 ----------
580 x : float
581 The x location of the text in display coords.
582 y : float
583 The y location of the text baseline in display coords.
584 s : str
585 The text to be converted.
586 prop : `~matplotlib.font_manager.FontProperties`
587 The font property.
588 angle : float
589 Angle in degrees to render the text at.
590 ismath : bool or "TeX"
591 If True, use mathtext parser. If "TeX", use tex for rendering.
592 """
593  
594 text2path = self._text2path
595 fontsize = self.points_to_pixels(prop.get_size_in_points())
596 verts, codes = text2path.get_text_path(prop, s, ismath=ismath)
597  
598 path = Path(verts, codes)
599 angle = np.deg2rad(angle)
600 if self.flipy():
601 width, height = self.get_canvas_width_height()
602 transform = (Affine2D()
603 .scale(fontsize / text2path.FONT_SCALE)
604 .rotate(angle)
605 .translate(x, height - y))
606 else:
607 transform = (Affine2D()
608 .scale(fontsize / text2path.FONT_SCALE)
609 .rotate(angle)
610 .translate(x, y))
611  
612 return path, transform

642 """

643 Get the width, height, and descent (offset from the bottom to the baseline), in
