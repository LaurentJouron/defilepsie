335 def tex_cmdname(string):
336 """Return leading TeX command name and remainder of `string`.
337  
338 >>> tex_cmdname('mymacro2') # up to first non-letter
339 ('mymacro', '2')
340 >>> tex_cmdname('name 2') # strip trailing whitespace
341 ('name', '2')
342 >>> tex_cmdname('_2') # single non-letter character
343 ('_', '2')
344  
345 """
346 m = re.match(r'([a-zA-Z]+)[ \n]*(.*)', string, re.DOTALL)
347 if m is None:
348 m = re.match(r'(.?)(.*)', string, re.DOTALL)
349 return m.group(1), m.group(2)
350

357 # ('name', '_2')
358 # >>> tex_cmdname('name_2\nnext line') # line-break allowed
359 # ('name', '_2\nnext line')

370 def tex_number(string):
371 """Return leading number literal and remainder of `string`.
372  
373 >>> tex_number('123.4')
374 ('123.4', '')
375  
376 """
377 m = re.match(r'([0-9.,]*[0-9]+)(.*)', string, re.DOTALL)
378 if m is None:
379 return '', string
380 return m.group(1), m.group(2)

397 def tex_token(string):
398 """Return first simple TeX token and remainder of `string`.
399  
400 >>> tex_token('\\command{without argument}')
401 ('\\command', '{without argument}')
402 >>> tex_token('or first character')
403 ('o', 'r first character')
404  
405 """
406 m = re.match(r"""((?P<cmd>\\[a-zA-Z]+)\s* # TeX command, skip whitespace
407 |(?P<chcmd>\\.) # one-character TeX command
408 |(?P<ch>.?)) # first character (or empty)
409 (?P<remainder>.*$) # remaining part of string
410 """, string, re.VERBOSE | re.DOTALL)
411 cmd, chcmd, ch, remainder = m.group('cmd', 'chcmd', 'ch', 'remainder')
412 return cmd or chcmd or ch, remainder

421 # ('.', ' but not after single char')
422 # >>> tex_token('') # empty string.
423 # ('', '')
424 # >>> tex_token('\{escaped bracket')
425 # ('\\{', 'escaped bracket')

428 def tex_group(string):
429 """Return first TeX group or token and remainder of `string`.
430  
431 >>> tex_group('{first group} returned without brackets')
432 ('first group', ' returned without brackets')
433  
434 """
435 split_index = 0
436 nest_level = 0 # level of {{nested} groups}
437 escape = False # the next character is escaped (\)
438  
439 if not string.startswith('{'):
440 # special case: there is no group, return first token and remainder
441 return string[:1], string[1:]
442 for c in string:
443 split_index += 1
444 if escape:
445 escape = False
446 elif c == '\\':
447 escape = True
448 elif c == '{':
449 nest_level += 1
450 elif c == '}':
451 nest_level -= 1
452 if nest_level == 0:
453 break
454 else:
455 raise MathError('Group without closing bracket!')
456 return string[1:split_index-1], string[split_index:]
457

460 # ('', ' empty group')
461 # >>> tex_group('{group with {nested} group} ')
462 # ('group with {nested} group', ' ')
463 # >>> tex_group('{group with {nested group}} at the end')
464 # ('group with {nested group}', ' at the end')
465 # >>> tex_group('{{group} {with {{complex }nesting}} constructs}')
466 # ('{group} {with {{complex }nesting}} constructs', '')
467 # >>> tex_group('{group with \\{escaped\\} brackets}')
468 # ('group with \\{escaped\\} brackets', '')
469 # >>> tex_group('{group followed by closing bracket}} from outer group')
470 # ('group followed by closing bracket', '} from outer group')
471 # >>> tex_group('No group? Return first character.')
472 # ('N', 'o group? Return first character.')
473 # >>> tex_group(' {also whitespace}')
474 # (' ', '{also whitespace}')

477 def tex_token_or_group(string):
478 """Return first TeX group or token and remainder of `string`.
479  
480 >>> tex_token_or_group('\\command{without argument}')
481 ('\\command', '{without argument}')
482 >>> tex_token_or_group('first character')
483 ('f', 'irst character')
484 >>> tex_token_or_group(' also whitespace')
485 (' ', 'also whitespace')
486 >>> tex_token_or_group('{first group} keep rest')
487 ('first group', ' keep rest')
488  
489 """
490 arg, remainder = tex_token(string)
491 if arg == '{':
492 arg, remainder = tex_group(string.lstrip())
493 return arg, remainder
494

496 # ('\\{', 'no group but left bracket')

532 def parse_latex_math(root, source):
533 """Append MathML conversion of `string` to `node` and return it.
534  
535 >>> parse_latex_math(math(), r'\alpha')
536 math(mi('α'))
537 >>> parse_latex_math(mrow(), r'x_{n}')
538 mrow(msub(mi('x'), mi('n')))
539  
540 """
541 # Normalize white-space:
542 string = source # not-yet handled part of source
543 node = root # the current "insertion point"
544  
545 # Loop over `string` while changing it.
546 while len(string) > 0:
547 # Take off first character:
548 c, string = string[0], string[1:]
549  
550 if c in ' \n':
551 continue # whitespace is ignored in LaTeX math mode
552 if c == '\\': # start of a LaTeX macro
553 cmdname, string = tex_cmdname(string)
554 node, string = handle_cmd(cmdname, node, string)
555 elif c in "_^":
556 node = handle_script_or_limit(node, c)
557 elif c == '{':
558 if isinstance(node, MathRow) and node.nchildren == 1:
559 # LaTeX takes one arg, MathML node accepts a group
560 node.nchildren = None # allow appending until closed by '}'
561 else: # wrap group in an <mrow>
562 new_node = mrow()
563 node.append(new_node)
564 node = new_node
565 elif c == '}':
566 node = node.close()
567 elif c == '&':
568 new_node = mtd()
569 node.close().append(new_node)
570 node = new_node
571 elif c.isalpha():
572 node = node.append(mi(c))
573 elif c.isdigit():
574 number, string = tex_number(string)
575 node = node.append(mn(c+number))
576 elif c in anomalous_chars:
577 # characters with a special meaning in LaTeX math mode
578 # fix spacing before "unary" minus.
579 attributes = {}
580 if c == '-' and len(node):
581 previous_node = node[-1]
582 if (previous_node.text and previous_node.text in '([='
  • E129 Visually indented line with same indent as next logical line
583 or previous_node.get('class') == 'mathopen'):
584 attributes['form'] = 'prefix'
585 node = node.append(mo(anomalous_chars[c], **attributes))
586 elif c in "/()[]|":
587 node = node.append(mo(c, stretchy=False))
588 elif c in "+*=<>,.!?`';@":
589 node = node.append(mo(c))
590 else:
591 raise MathError(f'Unsupported character: "{c}"!')
592 # TODO: append as <mi>?
593 if node is None:
594 if not string:
595 return root # ignore unbalanced braces
596 raise MathError(f'No insertion point for "{string}". '
597 f'Unbalanced braces in "{source[:-len(string)]}"?')
598 if node.nchildren and len(node) < node.nchildren:
599 raise MathError('Last node missing children. Source incomplete?')
600 return root
