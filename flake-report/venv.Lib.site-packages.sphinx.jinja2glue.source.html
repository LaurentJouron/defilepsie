57 def _slice_index(values: list, slices: int) -> Iterator[list]:
58 seq = values.copy()
59 length = 0
60 for value in values:
61 length += 1 + len(value[1][1]) # count includes subitems
62 items_per_slice = length // slices
63 offset = 0
64 for slice_number in range(slices):
65 count = 0
66 start = offset
67 if slices == slice_number + 1: # last column
68 offset = len(seq) # NoQA: SIM113
69 else:
70 for value in values[offset:]:
71 count += 1 + len(value[1][1])
72 offset += 1
73 if count >= items_per_slice:
74 break
75 yield seq[start:offset]
