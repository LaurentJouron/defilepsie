278 def set_aspect(self, aspect, adjustable=None, anchor=None, share=False):
279 """
280 Set the aspect ratios.
281  
282 Parameters
283 ----------
284 aspect : {'auto', 'equal', 'equalxy', 'equalxz', 'equalyz'}
285 Possible values:
286  
287 ========= ==================================================
288 value description
289 ========= ==================================================
290 'auto' automatic; fill the position rectangle with data.
291 'equal' adapt all the axes to have equal aspect ratios.
292 'equalxy' adapt the x and y axes to have equal aspect ratios.
293 'equalxz' adapt the x and z axes to have equal aspect ratios.
294 'equalyz' adapt the y and z axes to have equal aspect ratios.
295 ========= ==================================================
296  
297 adjustable : None or {'box', 'datalim'}, optional
298 If not *None*, this defines which parameter will be adjusted to
299 meet the required aspect. See `.set_adjustable` for further
300 details.
301  
302 anchor : None or str or 2-tuple of float, optional
303 If not *None*, this defines where the Axes will be drawn if there
304 is extra space due to aspect constraints. The most common way to
305 specify the anchor are abbreviations of cardinal directions:
306  
307 ===== =====================
308 value description
309 ===== =====================
310 'C' centered
311 'SW' lower left corner
312 'S' middle of bottom edge
313 'SE' lower right corner
314 etc.
315 ===== =====================
316  
317 See `~.Axes.set_anchor` for further details.
318  
319 share : bool, default: False
320 If ``True``, apply the settings to all shared Axes.
321  
322 See Also
323 --------
324 mpl_toolkits.mplot3d.axes3d.Axes3D.set_box_aspect
325 """
326 _api.check_in_list(('auto', 'equal', 'equalxy', 'equalyz', 'equalxz'),
327 aspect=aspect)
328 super().set_aspect(
329 aspect='auto', adjustable=adjustable, anchor=anchor, share=share)
330 self._aspect = aspect
331  
332 if aspect in ('equal', 'equalxy', 'equalxz', 'equalyz'):
333 ax_indices = self._equal_aspect_axis_indices(aspect)
334  
335 view_intervals = np.array([self.xaxis.get_view_interval(),
336 self.yaxis.get_view_interval(),
337 self.zaxis.get_view_interval()])
338 ptp = np.ptp(view_intervals, axis=1)
339 if self._adjustable == 'datalim':
340 mean = np.mean(view_intervals, axis=1)
341 scale = max(ptp[ax_indices] / self._box_aspect[ax_indices])
342 deltas = scale * self._box_aspect
343  
344 for i, set_lim in enumerate((self.set_xlim3d,
345 self.set_ylim3d,
346 self.set_zlim3d)):
347 if i in ax_indices:
348 set_lim(mean[i] - deltas[i]/2., mean[i] + deltas[i]/2.)
349 else: # 'box'
350 # Change the box aspect such that the ratio of the length of
351 # the unmodified axis to the length of the diagonal
352 # perpendicular to it remains unchanged.
353 box_aspect = np.array(self._box_aspect)
354 box_aspect[ax_indices] = ptp[ax_indices]
355 remaining_ax_indices = {0, 1, 2}.difference(ax_indices)
356 if remaining_ax_indices:
357 remaining = remaining_ax_indices.pop()
358 old_diag = np.linalg.norm(self._box_aspect[ax_indices])
359 new_diag = np.linalg.norm(box_aspect[ax_indices])
360 box_aspect[remaining] *= new_diag / old_diag
361 self.set_box_aspect(box_aspect)

384 def set_box_aspect(self, aspect, *, zoom=1):
385 """
386 Set the Axes box aspect.
387  
388 The box aspect is the ratio of height to width in display
389 units for each face of the box when viewed perpendicular to
390 that face. This is not to be confused with the data aspect (see
391 `~.Axes3D.set_aspect`). The default ratios are 4:4:3 (x:y:z).
392  
393 To simulate having equal aspect in data space, set the box
394 aspect to match your data range in each dimension.
395  
396 *zoom* controls the overall size of the Axes3D in the figure.
397  
398 Parameters
399 ----------
400 aspect : 3-tuple of floats or None
401 Changes the physical dimensions of the Axes3D, such that the ratio
402 of the axis lengths in display units is x:y:z.
403 If None, defaults to (4, 4, 3).
404  
405 zoom : float, default: 1
406 Control overall size of the Axes3D in the figure. Must be > 0.
407 """
408 if zoom <= 0:
409 raise ValueError(f'Argument zoom = {zoom} must be > 0')

412 aspect = np.asarray((4, 4, 3), dtype=float)
413 else:
414 aspect = np.asarray(aspect, dtype=float)
415 _api.check_shape((3,), aspect=aspect)
416 # default scale tuned to match the mpl32 appearance.
417 aspect *= 1.8294640721620434 * zoom / np.linalg.norm(aspect)

420 self.stale = True

629 function applies to 3D Axes, it also takes a *scalez* argument.
630 """
631 # This method looks at the rectangular volume (see above)
632 # of data and decides how to scale the view portal to fit it.
633 if tight is None:
634 _tight = self._tight
635 if not _tight:
636 # if image data only just use the datalim
637 for artist in self._children:
638 if isinstance(artist, mimage.AxesImage):
639 _tight = True
640 elif isinstance(artist, (mlines.Line2D, mpatches.Patch)):
641 _tight = False
642 break
643 else:
644 _tight = self._tight = bool(tight)
645  
646 if scalex and self.get_autoscalex_on():
647 x0, x1 = self.xy_dataLim.intervalx
648 xlocator = self.xaxis.get_major_locator()
649 x0, x1 = xlocator.nonsingular(x0, x1)
650 if self._xmargin > 0:
651 delta = (x1 - x0) * self._xmargin
652 x0 -= delta

896 def get_proj(self):
897 """Create the projection matrix from the current viewing position."""
898  
899 # Transform to uniform world coordinates 0-1, 0-1, 0-1
900 box_aspect = self._roll_to_vertical(self._box_aspect)
901 worldM = proj3d.world_transformation(
902 *self.get_xlim3d(),
903 *self.get_ylim3d(),
904 *self.get_zlim3d(),
905 pb_aspect=box_aspect,
906 )
907  
908 # Look into the middle of the world coordinates:
909 R = 0.5 * box_aspect
910  
911 # elev: elevation angle in the z plane.
912 # azim: azimuth angle in the xy plane.
913 # Coordinates for a point that rotates around the box of data.
914 # p0, p1 corresponds to rotating the box only around the vertical axis.
915 # p2 corresponds to rotating the box only around the horizontal axis.
916 elev_rad = np.deg2rad(self.elev)
917 azim_rad = np.deg2rad(self.azim)
918 p0 = np.cos(elev_rad) * np.cos(azim_rad)
919 p1 = np.cos(elev_rad) * np.sin(azim_rad)
920 p2 = np.sin(elev_rad)
921  
922 # When changing vertical axis the coordinates changes as well.
923 # Roll the values to get the same behaviour as the default:
924 ps = self._roll_to_vertical([p0, p1, p2])
925  
926 # The coordinates for the eye viewing point. The eye is looking
927 # towards the middle of the box of data from a distance:
928 eye = R + self._dist * ps
929  
930 # vvec, self._vvec and self._eye are unused, remove when deprecated
931 vvec = R - eye
932 self._eye = eye

934  
935 # Calculate the viewing axes for the eye position
936 u, v, w = self._calc_view_axes(eye)
937 self._view_u = u # _view_u is towards the right of the screen
938 self._view_v = v # _view_v is towards the top of the screen
939 self._view_w = w # _view_w is out of the screen
940  
941 # Generate the view and projection transformation matrices
942 if self._focal_length == np.inf:
943 # Orthographic projection
944 viewM = proj3d._view_transformation_uvw(u, v, w, eye)
945 projM = proj3d._ortho_transformation(-self._dist, self._dist)
946 else:
947 # Perspective projection
948 # Scale the eye dist to compensate for the focal length zoom effect
949 eye_focal = R + self._dist * ps * self._focal_length
950 viewM = proj3d._view_transformation_uvw(u, v, w, eye_focal)
951 projM = proj3d._persp_transformation(-self._dist,
952 self._dist,
953 self._focal_length)
954  
955 # Combine all the transformation matrices to get the final projection
956 M0 = np.dot(viewM, worldM)
957 M = np.dot(projM, M0)
958 return M
