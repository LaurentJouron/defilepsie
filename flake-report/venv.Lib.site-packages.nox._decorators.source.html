43 def _copy_func(src: T, name: str | None = None) -> T:
44 """This function copies another function, optionally with a new name."""
45  
46 dst = types.FunctionType(
47 src.__code__,
48 src.__globals__,
49 name=name or src.__name__,
50 argdefs=src.__defaults__,
51 closure=src.__closure__,
52 )
53 dst.__dict__.update(copy.deepcopy(src.__dict__))
54 dst = functools.update_wrapper(dst, src)
55 dst.__kwdefaults__ = src.__kwdefaults__
56 return cast(T, dst)
