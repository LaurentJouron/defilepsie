206 class TransformBlob:
207 def __init__(self):
208 self.identity = mtransforms.IdentityTransform()
209 self.identity2 = mtransforms.IdentityTransform()
210 # Force use of the more complex composition.
211 self.composite = mtransforms.CompositeGenericTransform(
212 self.identity,
213 self.identity2)
214 # Check parent -> child links of TransformWrapper.
215 self.wrapper = mtransforms.TransformWrapper(self.composite)
216 # Check child -> parent links of TransformWrapper.
217 self.composite2 = mtransforms.CompositeGenericTransform(
218 self.wrapper,
219 self.identity)
