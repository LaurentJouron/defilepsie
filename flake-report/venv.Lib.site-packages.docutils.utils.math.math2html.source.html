44 "Show a debug message"

50 "Show a trace message"
51 if Trace.quietmode:
52 return

54 message = Trace.prefix + message

57 def error(cls, message):
58 "Show an error message"
59 message = '* ' + message
60 if Trace.prefix and Trace.showlinesmode:
61 message = Trace.prefix + message
62 Trace.show(message, sys.stderr)

65 "Show a message out of a channel"
66 channel.write(message + '\n')
