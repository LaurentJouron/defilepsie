18 class PolarTransform(mtransforms.Transform):
19 r"""
20 The base polar transform.
21  
22 This transform maps polar coordinates :math:`\theta, r` into Cartesian
23 coordinates :math:`x, y = r \cos(\theta), r \sin(\theta)`
24 (but does not fully transform into Axes coordinates or
25 handle positioning in screen space).
26  
27 This transformation is designed to be applied to data after any scaling
28 along the radial axis (e.g. log-scaling) has been applied to the input
29 data.
30  
31 Path segments at a fixed radius are automatically transformed to circular
32 arcs as long as ``path._interpolation_steps > 1``.
33 """

81 # docstring inherited
82 if not len(path) or path._interpolation_steps == 1:
83 return Path(self.transform_non_affine(path.vertices), path.codes)
84 xys = []
85 codes = []
86 last_t = last_r = None
87 for trs, c in path.iter_segments():
88 trs = trs.reshape((-1, 2))

92 xys.extend(self.transform_non_affine(trs))
93 codes.append(Path.LINETO)

97 # that behavior here.
98 last_td, td = np.rad2deg([last_t, t])

100 r = ((r - self._get_rorigin())
101 * self._axis.get_rsign())
102 if last_td <= td:
103 while td - last_td > 360:
104 arc = Path.arc(last_td, last_td + 360)
105 xys.extend(arc.vertices[1:] * r)
106 codes.extend(arc.codes[1:])
107 last_td += 360
108 arc = Path.arc(last_td, td)
109 xys.extend(arc.vertices[1:] * r)
110 codes.extend(arc.codes[1:])
111 else:
112 # The reverse version also relies on the fact that all
113 # codes but the first one are the same.
114 while last_td - td > 360:
115 arc = Path.arc(last_td - 360, last_td)
116 xys.extend(arc.vertices[::-1][1:] * r)
117 codes.extend(arc.codes[1:])
118 last_td -= 360
119 arc = Path.arc(td, last_td)
120 xys.extend(arc.vertices[::-1][1:] * r)
121 codes.extend(arc.codes[1:])
122 else: # Interpolate.
123 trs = cbook.simple_linear_interpolation(
124 np.vstack([(last_t, last_r), trs]),
125 path._interpolation_steps)[1:]
126 xys.extend(self.transform_non_affine(trs))
127 codes.extend([Path.LINETO] * len(trs))
128 else: # Not a straight line.
129 xys.extend(self.transform_non_affine(trs))
130 codes.extend([c] * len(trs))
131 last_t, last_r = trs[-1]

140 class PolarAffine(mtransforms.Affine2DBase):
141 r"""
142 The affine part of the polar projection.
143  
144 Scales the output so that maximum radius rests on the edge of the axes
145 circle and the origin is mapped to (0.5, 0.5). The transform applied is
146 the same to x and y components and given by:
147  
148 .. math::
149  
150 x_{1} = 0.5 \left [ \frac{x_{0}}{(r_{\max} - r_{\min})} + 1 \right ]
151  
152 :math:`r_{\min}, r_{\max}` are the minimum and maximum radial limits after
153 any scaling (e.g. log scaling) has been removed.
154 """

157 Parameters
158 ----------
159 scale_transform : `~matplotlib.transforms.Transform`
160 Scaling transform for the data. This is used to remove any scaling
161 from the radial view limits.
162 limits : `~matplotlib.transforms.BboxBase`
163 View limits of the data. The only part of its bounds that is used
164 is the y limits (for the radius limits).
165 """
166 super().__init__()
167 self._scale_transform = scale_transform
168 self._limits = limits
169 self.set_children(scale_transform, limits)
170 self._mtx = None
171  
172 __str__ = mtransforms._make_str_method("_scale_transform", "_limits")

174 def get_matrix(self):
175 # docstring inherited
176 if self._invalid:

748 class _WedgeBbox(mtransforms.Bbox):
749 """
750 Transform (theta, r) wedge Bbox into axes bounding box.
751  
752 Parameters
753 ----------
754 center : (float, float)
755 Center of the wedge
756 viewLim : `~matplotlib.transforms.Bbox`
757 Bbox determining the boundaries of the wedge
758 originLim : `~matplotlib.transforms.Bbox`
759 Bbox determining the origin for the wedge, if different from *viewLim*
760 """
761 def __init__(self, center, viewLim, originLim, **kwargs):
762 super().__init__([[0, 0], [1, 1]], **kwargs)
763 self._center = center
764 self._viewLim = viewLim
765 self._originLim = originLim
766 self.set_children(viewLim, originLim)
767  
768 __str__ = mtransforms._make_str_method("_center", "_viewLim", "_originLim")
769  
770 def get_points(self):
771 # docstring inherited
772 if self._invalid:
773 points = self._viewLim.get_points().copy()
774 # Scale angular limits to work with Wedge.
775 points[:, 0] *= 180 / np.pi
776 if points[0, 0] > points[1, 0]:
777 points[:, 0] = points[::-1, 0]
778  
779 # Scale radial limits based on origin radius.
780 points[:, 1] -= self._originLim.y0
781  
782 # Scale radial limits to match axes limits.
783 rscale = 0.5 / points[1, 1]
784 points[:, 1] *= rscale
785 width = min(points[1, 1] - points[0, 1], 0.5)
786  
787 # Generate bounding box for wedge.
788 wedge = mpatches.Wedge(self._center, points[1, 1],
789 points[0, 0], points[1, 0],
790 width=width)
791 self.update_from_path(wedge.get_path())
792  
793 # Ensure equal aspect ratio.
794 w, h = self._points[1] - self._points[0]
795 deltah = max(w - h, 0) / 2
796 deltaw = max(h - w, 0) / 2
797 self._points += np.array([[-deltaw, -deltah], [deltaw, deltah]])
798  
799 self._invalid = 0
800  
801 return self._points

853 def _set_lim_and_transforms(self):
854 # A view limit where the minimum radius can be locked if the user
855 # specifies an alternate origin.
856 self._originViewLim = mtransforms.LockableBbox(self.viewLim)
857  
858 # Handle angular offset and direction.
859 self._direction = mtransforms.Affine2D() \
860 .scale(self._default_theta_direction, 1.0)
861 self._theta_offset = mtransforms.Affine2D() \
862 .translate(self._default_theta_offset, 0.0)
863 self.transShift = self._direction + self._theta_offset
864 # A view limit shifted to the correct location after accounting for
865 # orientation and offset.
866 self._realViewLim = mtransforms.TransformedBbox(self.viewLim,
867 self.transShift)
868  
869 # Transforms the x and y axis separately by a scale factor
870 # It is assumed that this part will have non-linear components
871 self.transScale = mtransforms.TransformWrapper(
872 mtransforms.IdentityTransform())
873  
874 # Scale view limit into a bbox around the selected wedge. This may be
875 # smaller than the usual unit axes rectangle if not plotting the full
876 # circle.
877 self.axesLim = _WedgeBbox((0.5, 0.5),
878 self._realViewLim, self._originViewLim)
879  
880 # Scale the wedge to fill the axes.
881 self.transWedge = mtransforms.BboxTransformFrom(self.axesLim)
882  
883 # Scale the axes to fill the figure.
884 self.transAxes = mtransforms.BboxTransformTo(self.bbox)
885  
886 # A (possibly non-linear) projection on the (already scaled)
887 # data. This one is aware of rmin
888 self.transProjection = self.PolarTransform(
889 self,
890 _apply_theta_transforms=False,
891 scale_transform=self.transScale
892 )
893 # Add dependency on rorigin.
894 self.transProjection.set_children(self._originViewLim)
895  
896 # An affine transformation on the data, generally to limit the
897 # range of the axes
898 self.transProjectionAffine = self.PolarAffine(self.transScale,
899 self._originViewLim)
900  
901 # The complete data transformation stack -- from data all the
902 # way to display coordinates
903 #
904 # 1. Remove any radial axis scaling (e.g. log scaling)
905 # 2. Shift data in the theta direction
906 # 3. Project the data from polar to cartesian values
907 # (with the origin in the same place)
908 # 4. Scale and translate the cartesian values to Axes coordinates
909 # (here the origin is moved to the lower left of the Axes)
910 # 5. Move and scale to fill the Axes
911 # 6. Convert from Axes coordinates to Figure coordinates
912 self.transData = (
913 self.transScale +
914 self.transShift +
915 self.transProjection +
916 (
917 self.transProjectionAffine +
918 self.transWedge +
919 self.transAxes
920 )
921 )
922  
923 # This is the transform for theta-axis ticks. It is
924 # equivalent to transData, except it always puts r == 0.0 and r == 1.0
925 # at the edge of the axis circles.
926 self._xaxis_transform = (
927 mtransforms.blended_transform_factory(
928 mtransforms.IdentityTransform(),
929 mtransforms.BboxTransformTo(self.viewLim)) +
930 self.transData)
931 # The theta labels are flipped along the radius, so that text 1 is on
932 # the outside by default. This should work the same as before.
933 flipr_transform = mtransforms.Affine2D() \
934 .translate(0.0, -0.5) \
935 .scale(1.0, -1.0) \
936 .translate(0.0, 0.5)
937 self._xaxis_text_transform = flipr_transform + self._xaxis_transform
938  
939 # This is the transform for r-axis ticks. It scales the theta
940 # axis so the gridlines from 0.0 to 1.0, now go from thetamin to
941 # thetamax.
942 self._yaxis_transform = (
943 mtransforms.blended_transform_factory(
944 mtransforms.BboxTransformTo(self.viewLim),
945 mtransforms.IdentityTransform()) +
946 self.transData)
947 # The r-axis labels are put at an angle and padded in the r-direction
948 self._r_label_position = mtransforms.Affine2D() \
949 .translate(self._default_rlabel_position, 0.0)
950 self._yaxis_text_transform = mtransforms.TransformWrapper(
951 self._r_label_position + self.transData)

1070 def set_thetalim(self, *args, **kwargs):
1071 r"""
1072 Set the minimum and maximum theta values.
1073  
1074 Can take the following signatures:
1075  
1076 - ``set_thetalim(minval, maxval)``: Set the limits in radians.
1077 - ``set_thetalim(thetamin=minval, thetamax=maxval)``: Set the limits
1078 in degrees.
1079  
1080 where minval and maxval are the minimum and maximum limits. Values are
1081 wrapped in to the range :math:`[0, 2\pi]` (in radians), so for example
1082 it is possible to do ``set_thetalim(-np.pi / 2, np.pi / 2)`` to have
1083 an axis symmetric around 0. A ValueError is raised if the absolute
1084 angle difference is larger than a full circle.
1085 """
1086 orig_lim = self.get_xlim() # in radians
1087 if 'thetamin' in kwargs:
1088 kwargs['xmin'] = np.deg2rad(kwargs.pop('thetamin'))
1089 if 'thetamax' in kwargs:
1090 kwargs['xmax'] = np.deg2rad(kwargs.pop('thetamax'))
1091 new_min, new_max = self.set_xlim(*args, **kwargs)
1092 # Parsing all permutations of *args, **kwargs is tricky; it is simpler
1093 # to let set_xlim() do it and then validate the limits.
1094 if abs(new_max - new_min) > 2 * np.pi:
1095 self.set_xlim(orig_lim) # un-accept the change
1096 raise ValueError("The angle range must be less than a full circle")
1097 return tuple(np.rad2deg((new_min, new_max)))

1416 def format_coord(self, theta, r):
1417 # docstring inherited
1418 screen_xy = self.transData.transform((theta, r))
1419 screen_xys = screen_xy + np.stack(
1420 np.meshgrid([-1, 0, 1], [-1, 0, 1])).reshape((2, -1)).T
1421 ts, rs = self.transData.inverted().transform(screen_xys).T
1422 delta_t = abs((ts - theta + np.pi) % (2 * np.pi) - np.pi).max()
1423 delta_t_halfturns = delta_t / np.pi
1424 delta_t_degrees = delta_t_halfturns * 180
1425 delta_r = abs(rs - r).max()
1426 if theta < 0:
1427 theta += 2 * np.pi
1428 theta_halfturns = theta / np.pi
1429 theta_degrees = theta_halfturns * 180
1430  
1431 # See ScalarFormatter.format_data_short. For r, use #g-formatting
1432 # (as for linear axes), but for theta, use f-formatting as scientific
1433 # notation doesn't make sense and the trailing dot is ugly.
1434 def format_sig(value, delta, opt, fmt):
1435 # For "f", only count digits after decimal point.
1436 prec = (max(0, -math.floor(math.log10(delta))) if fmt == "f" else
1437 cbook._g_sig_digits(value, delta))
1438 return f"{value:-{opt}.{prec}{fmt}}"
1439  
1440 return ('\N{GREEK SMALL LETTER THETA}={}\N{GREEK SMALL LETTER PI} '
1441 '({}\N{DEGREE SIGN}), r={}').format(
1442 format_sig(theta_halfturns, delta_t_halfturns, "", "f"),
1443 format_sig(theta_degrees, delta_t_degrees, "", "f"),
1444 format_sig(r, delta_r, "#", "g"),
1445 )

1478 epsilon = np.pi / 45.0
1479 t, r = self.transData.inverted().transform((x, y))

1481 mode = 'drag_r_labels'
1482 elif button == 3:

1497 def drag_pan(self, button, key, x, y):
1498 p = self._pan_start
1499  
1500 if p.mode == 'drag_r_labels':
1501 (startt, startr), (t, r) = p.trans_inverse.transform(
1502 [(p.x, p.y), (x, y)])
1503  
1504 # Deal with theta
1505 dt = np.rad2deg(startt - t)
1506 self.set_rlabel_position(p.r_label_angle - dt)

1508 trans, vert1, horiz1 = self.get_yaxis_text1_transform(0.0)
1509 trans, vert2, horiz2 = self.get_yaxis_text2_transform(0.0)

1511 t.label1.set_va(vert1)

1516 elif p.mode == 'zoom':
1517 (startt, startr), (t, r) = p.trans_inverse.transform(
1518 [(p.x, p.y), (x, y)])
