44 def is_allowed_version(spec: str, version: str) -> bool:
45 """Check `spec` satisfies `version` or not.
46  
47 This obeys PEP-440 specifiers:
48 https://peps.python.org/pep-0440/#version-specifiers
49  
50 Some examples:
51  
52 >>> is_allowed_version('<=3.5', '3.3')
53 True
54 >>> is_allowed_version('<=3.2', '3.3')
55 False
56 >>> is_allowed_version('>3.2, <4.0', '3.3')
57 True
58 """
59 return Version(version) in SpecifierSet(spec)
