177 def sort_errors(results, sorted_by):
178 """Flattens and potentially sorts errors based on user prefernces"""
179 if not sorted_by:
180 for results in results:
181 yield from results
182 return
183 errors = list(error for errors in results for error in errors)
184 # sorting is stable in python, so we can sort in reverse order to get the
185 # ordering specified by the user
186 for sort_field in reversed(sorted_by):
187 if sort_field == SortField.ERROR_TYPE:
188 errors.sort(key=lambda error: error.checker_name)
189 elif sort_field == SortField.FILENAME:
190 errors.sort(key=lambda error: error.filename)
191 elif sort_field == SortField.LINE:
192 errors.sort(key=lambda error: error.line_no)
193 yield from errors
