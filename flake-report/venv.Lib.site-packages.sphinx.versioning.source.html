33 def add_uids(doctree: Node, condition: Any) -> Iterator[Node]:
34 """Add a unique id to every node in the `doctree` which matches the

48 def merge_doctrees(old: Node, new: Node, condition: Any) -> Iterator[Node]:
49 """Merge the `old` doctree with the `new` one while looking at nodes
50 matching the `condition`.
51  
52 Each node which replaces another one or has been added to the `new` doctree
53 will be yielded.
54  
55 :param condition:
56 A callable which returns either ``True`` or ``False`` for a given node.
57 """
58 old_iter = old.findall(condition)
59 new_iter = new.findall(condition)
60 old_nodes = []
61 new_nodes = []
62 ratios = {}
63 seen = set()
64 # compare the nodes each doctree in order
65 for old_node, new_node in zip_longest(old_iter, new_iter):
66 if old_node is None:
67 new_nodes.append(new_node)
68 continue
69 if not getattr(old_node, 'uid', None):
70 # maybe config.gettext_uuid has been changed.
71 old_node.uid = uuid4().hex
72 if new_node is None:
73 old_nodes.append(old_node)
74 continue
75 ratio = get_ratio(old_node.rawsource, new_node.rawsource)
76 if ratio == 0:
77 new_node.uid = old_node.uid
78 seen.add(new_node)
79 else:
80 ratios[old_node, new_node] = ratio
81 old_nodes.append(old_node)
82 new_nodes.append(new_node)
83 # calculate the ratios for each unequal pair of nodes, should we stumble
84 # on a pair which is equal we set the uid and add it to the seen ones
85 for old_node, new_node in product(old_nodes, new_nodes):
86 if new_node in seen or (old_node, new_node) in ratios:
87 continue
88 ratio = get_ratio(old_node.rawsource, new_node.rawsource)
89 if ratio == 0:
90 new_node.uid = old_node.uid
91 seen.add(new_node)
92 else:
93 ratios[old_node, new_node] = ratio

95 # as long as the ratio is under a certain value, in which case we consider
96 # them not changed but different

97 ratios = sorted(ratios.items(), key=itemgetter(1)) # type: ignore[assignment]
98 for (old_node, new_node), ratio in ratios:
99 if new_node in seen:
100 continue
101 else:
102 seen.add(new_node)
103 if ratio < VERSIONING_RATIO:
104 new_node.uid = old_node.uid
105 else:
106 new_node.uid = uuid4().hex

108 # create new uuids for any new node we left out earlier, this happens

128 def levenshtein_distance(a: str, b: str) -> int:
129 """Return the Levenshtein edit distance between two strings *a* and *b*."""
130 if a == b:
131 return 0
132 if len(a) < len(b):
133 a, b = b, a
134 if not a:
135 return len(b)
136 previous_row = list(range(len(b) + 1))
137 for i, column1 in enumerate(a):
138 current_row = [i + 1]
139 for j, column2 in enumerate(b):
140 insertions = previous_row[j + 1] + 1
141 deletions = current_row[j] + 1
142 substitutions = previous_row[j] + (column1 != column2)
143 current_row.append(min(insertions, deletions, substitutions))
144 previous_row = current_row
145 return previous_row[-1]
